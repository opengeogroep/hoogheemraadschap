 20 /**
 21  * Tests whether a triangle is acute. A triangle is acute iff all interior
 22  * angles are acute. This is a strict test - right triangles will return
 23  * <tt>false</tt> A triangle which is not acute is either right or obtuse.
 24  * <p>
 25  * Note: this implementation is not robust for angles very close to 90 degrees.
 26  *
 27  * @param {jsts.geom.Coordinate}
 28  *          a a vertex of the triangle.
 29  * @param {jsts.geom.Coordinate}
 30  *          b a vertex of the triangle.
 31  * @param {jsts.geom.Coordinate}
 32  *          c a vertex of the triangle.
 33  * @return {Boolean} true if the triangle is acute.
 34  */
 35 jsts.geom.Triangle.isAcute = function(a, b, c) {
 36   if (!jsts.algorithm.Angle.isAcute(a, b, c)) {
 37     return false;
 38   }
 39   if (!jsts.algorithm.Angle.isAcute(b, c, a)) {
 40     return false;
 41   }
 42   if (!jsts.algorithm.Angle.isAcute(c, a, b)) {
 43     return false;
 44   }
 45   return true;
 46 };

 49 /**
 50  * Computes the line which is the perpendicular bisector of the line segment
 51  * a-b.
 52  *
 53  * @param {jsts.geom.Coordinate}
 54  *          a a point.
 55  * @param {jsts.geom.Coordinate}
 56  *          b another point.
 57  * @return {jsts.algorithm.HCoordinate} the perpendicular bisector, as an
 58  *         HCoordinate.
 59  */
 60 jsts.geom.Triangle.perpendicularBisector = function(a, b) {
 61   // returns the perpendicular bisector of the line segment ab
 62   var dx, dy, l1, l2;
 63 
 64   dx = b.x - a.x;
 65   dy = b.y - a.y;
 66 
 67   l1 = new jsts.algorithm.HCoordinate(a.x + dx / 2.0, a.y + dy / 2.0, 1.0);
 68   l2 = new jsts.algorithm.HCoordinate(a.x - dy + dx / 2.0, a.y + dx + dy / 2.0,
 69       1.0);
 70   return new jsts.algorithm.HCoordinate(l1, l2);
 71 };

 74 /**
 75  * Computes the circumcentre of a triangle. The circumcentre is the centre of
 76  * the circumcircle, the smallest circle which encloses the triangle. It is also
 77  * the common intersection point of the perpendicular bisectors of the sides of
 78  * the triangle, and is the only point which has equal distance to all three
 79  * vertices of the triangle.
 80  * <p>
 81  * This method uses an algorithm due to J.R.Shewchuk which uses normalization to
 82  * the origin to improve the accuracy of computation. (See <i>Lecture Notes on
 83  * Geometric Robustness</i>, Jonathan Richard Shewchuk, 1999).
 84  *
 85  * @param {jsts.geom.Coordinate}
 86  *          a a vertx of the triangle.
 87  * @param {jsts.geom.Coordinate}
 88  *          b a vertx of the triangle.
 89  * @param {jsts.geom.Coordinate}
 90  *          c a vertx of the triangle.
 91  * @return {jsts.geom.Coordinate} the circumcentre of the triangle.
 92  */
 93 jsts.geom.Triangle.circumcentre = function(a, b, c) {
 94   var cx, cy, ax, ay, bx, by, denom, numx, numy, ccx, ccy;
 95 
 96   cx = c.x;
 97   cy = c.y;
 98   ax = a.x - cx;
 99   ay = a.y - cy;
100   bx = b.x - cx;
101   by = b.y - cy;
102 
103   denom = 2 * jsts.geom.Triangle.det(ax, ay, bx, by);
104   numx = jsts.geom.Triangle.det(ay, ax * ax + ay * ay, by, bx * bx + by * by);
105   numy = jsts.geom.Triangle.det(ax, ax * ax + ay * ay, bx, bx * bx + by * by);
106 
107   ccx = cx - numx / denom;
108   ccy = cy + numy / denom;
109 
110   return new jsts.geom.Coordinate(ccx, ccy);
111 };

114 /**
115  * Computes the determinant of a 2x2 matrix. Uses standard double-precision
116  * arithmetic, so is susceptible to round-off error.
117  *
118  * @param {Number}
119  *          m00 the [0,0] entry of the matrix.
120  * @param {Number}
121  *          m01 the [0,1] entry of the matrix.
122  * @param {Number}
123  *          m10 the [1,0] entry of the matrix.
124  * @param {Number}
125  *          m11 the [1,1] entry of the matrix.
126  * @return {Number} the determinant.
127  */
128 jsts.geom.Triangle.det = function(m00, m01, m10, m11) {
129   return m00 * m11 - m01 * m10;
130 };

133 /**
134  * Computes the incentre of a triangle. The <i>inCentre</i> of a triangle is
135  * the point which is equidistant from the sides of the triangle. It is also the
136  * point at which the bisectors of the triangle's angles meet. It is the centre
137  * of the triangle's <i>incircle</i>, which is the unique circle that is
138  * tangent to each of the triangle's three sides.
139  *
140  * @param {jsts.geom.Coordinate}
141  *          a a vertx of the triangle.
142  * @param {jsts.geom.Coordinate}
143  *          b a vertx of the triangle.
144  * @param {jsts.geom.Coordinate}
145  *          c a vertx of the triangle.
146  * @return {jsts.geom.Coordinate} the point which is the incentre of the
147  *         triangle.
148  */
149 jsts.geom.Triangle.inCentre = function(a, b, c) {
150   var len0, len1, len2, circum, inCentreX, inCentreY;
151 
152   // the lengths of the sides, labelled by their opposite vertex
153   len0 = b.distance(c);
154   len1 = a.distance(c);
155   len2 = a.distance(b);
156   circum = len0 + len1 + len2;
157 
158   inCentreX = (len0 * a.x + len1 * b.x + len2 * c.x) / circum;
159   inCentreY = (len0 * a.y + len1 * b.y + len2 * c.y) / circum;
160 
161   return new jsts.geom.Coordinate(inCentreX, inCentreY);
162 };

165 /**
166  * Computes the centroid (centre of mass) of a triangle. This is also the point
167  * at which the triangle's three medians intersect (a triangle median is the
168  * segment from a vertex of the triangle to the midpoint of the opposite side).
169  * The centroid divides each median in a ratio of 2:1. The centroid always lies
170  * within the triangle.
171  *
172  *
173  * @param {jsts.geom.Coordinate}
174  *          a a vertx of the triangle.
175  * @param {jsts.geom.Coordinate}
176  *          b a vertx of the triangle.
177  * @param {jsts.geom.Coordinate}
178  *          c a vertx of the triangle.
179  * @return {jsts.geom.Coordinate} the centroid of the triangle.
180  */
181 jsts.geom.Triangle.centroid = function(a, b, c) {
182   var x, y;
183 
184   x = (a.x + b.x + c.x) / 3;
185   y = (a.y + b.y + c.y) / 3;
186 
187   return new jsts.geom.Coordinate(x, y);
188 };

191 /**
192  * Computes the length of the longest side of a triangle
193  *
194  * @param {jsts.geom.Coordinate}
195  *          a a vertex of the triangle.
196  * @param {jsts.geom.Coordinate}
197  *          b a vertex of the triangle.
198  * @param {jsts.geom.Coordinate}
199  *          c a vertex of the triangle.
200  * @return {Number} the length of the longest side of the triangle.
201  */
202 jsts.geom.Triangle.longestSideLength = function(a, b, c) {
203   var lenAB, lenBC, lenCA, maxLen;
204 
205   lenAB = a.distance(b);
206   lenBC = b.distance(c);
207   lenCA = c.distance(a);
208   maxLen = lenAB;
209 
210   if (lenBC > maxLen) {
211     maxLen = lenBC;
212   }
213   if (lenCA > maxLen) {
214     maxLen = lenCA;
215   }
216   return maxLen;
217 };

220 /**
221  * Computes the point at which the bisector of the angle ABC cuts the segment
222  * AC.
223  *
224  * @param {jsts.geom.Coordinate}
225  *          a a vertex of the triangle.
226  * @param {jsts.geom.Coordinate}
227  *          b a vertex of the triangle.
228  * @param {jsts.geom.Coordinate}
229  *          c a vertex of the triangle.
230  * @return {jsts.geom.Coordinate} the angle bisector cut point.
231  */
232 jsts.geom.Triangle.angleBisector = function(a, b, c) {
233   /**
234    * Uses the fact that the lengths of the parts of the split segment are
235    * proportional to the lengths of the adjacent triangle sides
236    */
237   var len0, len2, frac, dx, dy, splitPt;
238 
239   len0 = b.distance(a);
240   len2 = b.distance(c);
241   frac = len0 / (len0 + len2);
242   dx = c.x - a.x;
243   dy = c.y - a.y;
244 
245   splitPt = new jsts.geom.Coordinate(a.x + frac * dx, a.y + frac * dy);
246   return splitPt;
247 };

250 /**
251  * Computes the 2D area of a triangle. The area value is always non-negative.
252  *
253  * @param {jsts.geom.Coordinate}
254  *          a a vertex of the triangle.
255  * @param {jsts.geom.Coordinate}
256  *          b a vertex of the triangle.
257  * @param {jsts.geom.Coordinate}
258  *          c a vertex of the triangle.
259  * @return {Number} the area of the triangle.
260  */
261 jsts.geom.Triangle.area = function(a, b, c) {
262   return Math
263       .abs(((c.x - a.x) * (b.y - a.y) - (b.x - a.x) * (c.y - a.y)) / 2.0);
264 };

267 /**
268  * Computes the signed 2D area of a triangle. The area value is positive if the
269  * triangle is oriented CW, and negative if it is oriented CCW.
270  * <p>
271  * The signed area value can be used to determine point orientation, but the
272  * implementation in this method is susceptible to round-off errors. Use
273  * {@link CGAlgorithms#orientationIndex(Coordinate, Coordinate, Coordinate)} for
274  * robust orientation calculation.
275  *
276  * @param {jsts.geom.Coordinate}
277  *          a a vertex of the triangle.
278  * @param {jsts.geom.Coordinate}
279  *          b a vertex of the triangle.
280  * @param {jsts.geom.Coordinate}
281  *          c a vertex of the triangle.
282  * @return {Number} the signed 2D area of the triangle.
283  *
284  */
285 jsts.geom.Triangle.signedArea = function(a, b, c) {
286   /**
287    * Uses the formula 1/2 * | u x v | where u,v are the side vectors of the
288    * triangle x is the vector cross-product For 2D vectors, this formula
289    * simplifies to the expression below
290    */
291   return ((c.x - a.x) * (b.y - a.y) - (b.x - a.x) * (c.y - a.y)) / 2.0;
292 };
