  9 /**
 10  * A list of the {@link SegmentNode}s present along a noded
 11  * {@link SegmentString}.
 12  *
 13  * @constructor
 14  */
 15 jsts.noding.SegmentNodeList = function(edge) {
 16   this.nodeMap = new javascript.util.TreeMap();
 17 
 18   this.edge = edge;
 19 };
 20 
 21 
 22 /**
 23  * @type {javascript.util.TreeMap}
 24  * @private
 25  */
 26 jsts.noding.SegmentNodeList.prototype.nodeMap = null;
 27 
 28 /**
 29  * returns an iterator of SegmentNodes
 30  */
 31 jsts.noding.SegmentNodeList.prototype.iterator = function() {
 32   return this.nodeMap.values().iterator();
 33 };

 42 jsts.noding.SegmentNodeList.prototype.edge = null;

 45 jsts.noding.SegmentNodeList.prototype.getEdge = function() {

 50 /**
 51  * Adds an intersection into the list, if it isn't already there. The input
 52  * segmentIndex and dist are expected to be normalized.
 53  *
 54  * @return the SegmentIntersection found or added.
 55  */
 56 jsts.noding.SegmentNodeList.prototype.add = function(intPt, segmentIndex) {
 57   var eiNew = new jsts.noding.SegmentNode(this.edge, intPt, segmentIndex,
 58       this.edge.getSegmentOctant(segmentIndex));
 59   var ei = this.nodeMap.get(eiNew);
 60   if (ei !== null) {
 61     jsts.util.Assert.isTrue(ei.coord.equals2D(intPt),
 62         'Found equal nodes with different coordinates');
 63     return ei;
 64   }
 65   // node does not exist, so create it
 66   this.nodeMap.put(eiNew, eiNew);
 67   return eiNew;
 68 };

 71 /**
 72  * Adds nodes for the first and last points of the edge
 73  *
 74  * @private
 75  */
 76 jsts.noding.SegmentNodeList.prototype.addEndpoints = function() {
 77   var maxSegIndex = this.edge.size() - 1;
 78   this.add(this.edge.getCoordinate(0), 0);
 79   this.add(this.edge.getCoordinate(maxSegIndex), maxSegIndex);
 80 };

 83 /**
 84  * Adds nodes for any collapsed edge pairs. Collapsed edge pairs can be caused
 85  * by inserted nodes, or they can be pre-existing in the edge vertex list. In
 86  * order to provide the correct fully noded semantics, the vertex at the base of
 87  * a collapsed pair must also be added as a node.
 88  *
 89  * @private
 90  */
 91 jsts.noding.SegmentNodeList.prototype.addCollapsedNodes = function() {
 92   var collapsedVertexIndexes = [];
 93 
 94   this.findCollapsesFromInsertedNodes(collapsedVertexIndexes);
 95   this.findCollapsesFromExistingVertices(collapsedVertexIndexes);
 96 
 97   // node the collapses
 98   for (var i = 0; i < collapsedVertexIndexes.length; i++) {
 99     var vertexIndex = collapsedVertexIndexes[i];
100     this.add(this.edge.getCoordinate(vertexIndex), vertexIndex);
101   }
102 };

105 /**
106  * Adds nodes for any collapsed edge pairs which are pre-existing in the vertex
107  * list.
108  *
109  * @private
110  */
111 jsts.noding.SegmentNodeList.prototype.findCollapsesFromExistingVertices = function(
112     collapsedVertexIndexes) {
113   for (var i = 0; i < this.edge.size() - 2; i++) {
114     var p0 = this.edge.getCoordinate(i);
115     var p1 = this.edge.getCoordinate(i + 1);
116     var p2 = this.edge.getCoordinate(i + 2);
117     if (p0.equals2D(p2)) {
118       // add base of collapse as node
119       collapsedVertexIndexes.push(i + 1);
120     }
121   }
122 };
123 
124 
125 /**
126  * Adds nodes for any collapsed edge pairs caused by inserted nodes Collapsed
127  * edge pairs occur when the same coordinate is inserted as a node both before
128  * and after an existing edge vertex. To provide the correct fully noded
129  * semantics, the vertex must be added as a node as well.
130  *
131  * @private
132  */
133 jsts.noding.SegmentNodeList.prototype.findCollapsesFromInsertedNodes = function(
134     collapsedVertexIndexes) {
135   var collapsedVertexIndex = [null];
136   var it = this.iterator();
137   // there should always be at least two entries in the list, since the
138   // endpoints are nodes
139   var eiPrev = it.next();
140   while (it.hasNext()) {
141     var ei = it.next();
142     var isCollapsed = this.findCollapseIndex(eiPrev, ei, collapsedVertexIndex);
143     if (isCollapsed)
144       collapsedVertexIndexes.push(collapsedVertexIndex[0]);
145 
146     eiPrev = ei;
147   }
148 };
149 
150 
151 /**
152  * @private
153  */
154 jsts.noding.SegmentNodeList.prototype.findCollapseIndex = function(ei0, ei1,
155     collapsedVertexIndex) {
156   // only looking for equal nodes
157   if (!ei0.coord.equals2D(ei1.coord))
158     return false;
159 
160   var numVerticesBetween = ei1.segmentIndex - ei0.segmentIndex;
161   if (!ei1.isInterior()) {
162     numVerticesBetween--;
163   }
164 
165   // if there is a single vertex between the two equal nodes, this is a collapse
166   if (numVerticesBetween === 1) {
167     collapsedVertexIndex[0] = ei0.segmentIndex + 1;
168     return true;
169   }
170   return false;
171 };

174 /**
175  * Creates new edges for all the edges that the intersections in this list split
176  * the parent edge into. Adds the edges to the provided argument list (this is
177  * so a single list can be used to accumulate all split edges for a set of
178  * {@link SegmentString}s).
179  *
180  * @param {Array}
181  *          edgeList
182  */
183 jsts.noding.SegmentNodeList.prototype.addSplitEdges = function(edgeList) {
184   // ensure that the list has entries for the first and last point of the edge
185   this.addEndpoints();
186   this.addCollapsedNodes();
187 
188   var it = this.iterator();
189   // there should always be at least two entries in the list, since the
190   // endpoints are nodes
191   var eiPrev = it.next();
192   while (it.hasNext()) {
193     var ei = it.next();
194     var newEdge = this.createSplitEdge(eiPrev, ei);
195     edgeList.add(newEdge);
196     eiPrev = ei;
197   }
198 };

201 /**
202  * Checks the correctness of the set of split edges corresponding to this edge
203  *
204  * @param {Array}
205  *          splitEdges the split edges for this edge (in order).
206  * @private
207  */
208 jsts.noding.SegmentNodeList.prototype.checkSplitEdgesCorrectness = function(
209     splitEdges) {
210   var edgePts = edge.getCoordinates();
211 
212   // check that first and last points of split edges are same as endpoints of
213   // edge
214   var split0 = splitEdges[0];
215   var pt0 = split0.getCoordinate(0);
216   if (!pt0.equals2D(edgePts[0]))
217     throw new Error('bad split edge start point at ' + pt0);
218 
219   var splitn = splitEdges[splitEdges.length - 1];
220   var splitnPts = splitn.getCoordinates();
221   var ptn = splitnPts[splitnPts.length - 1];
222   if (!ptn.equals2D(edgePts[edgePts.length - 1]))
223     throw new Error('bad split edge end point at ' + ptn);
224 };

227 /**
228  * Create a new "split edge" with the section of points between (and including)
229  * the two intersections. The label for the new edge is the same as the label
230  * for the parent edge.
231  *
232  * @private
233  */
234 jsts.noding.SegmentNodeList.prototype.createSplitEdge = function(ei0, ei1) {
235   var npts = ei1.segmentIndex - ei0.segmentIndex + 2;
236 
237   var lastSegStartPt = this.edge.getCoordinate(ei1.segmentIndex);
238   // if the last intersection point is not equal to the its segment start pt,
239   // add it to the points list as well.
240   // (This check is needed because the distance metric is not totally reliable!)
241   // The check for point equality is 2D only - Z values are ignored
242   var useIntPt1 = ei1.isInterior() || !ei1.coord.equals2D(lastSegStartPt);
243   if (!useIntPt1) {
244     npts--;
245   }
246 
247   var pts = [];
248   // pts.length = npts;
249   var ipt = 0;
250   pts[ipt++] = new jsts.geom.Coordinate(ei0.coord);
251   for (var i = ei0.segmentIndex + 1; i <= ei1.segmentIndex; i++) {
252     pts[ipt++] = this.edge.getCoordinate(i);
253   }
254   if (useIntPt1)
255     pts[ipt] = ei1.coord;
256 
257   return new jsts.noding.NodedSegmentString(pts, this.edge.getData());
258 };
