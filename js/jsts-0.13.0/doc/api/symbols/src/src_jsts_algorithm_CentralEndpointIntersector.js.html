 58 /**
 59  * @private
 60  */
 61 jsts.algorithm.CentralEndpointIntersector.prototype.compute = function() {
 62   var centroid = jsts.algorithm.CentralEndpointIntersector.average(this.pts);
 63   this.intPt = this.findNearestPoint(centroid, this.pts);
 64 };

 67 /**
 68  * @return {Coordinate}
 69  */
 70 jsts.algorithm.CentralEndpointIntersector.prototype.getIntersection = function() {
 71   return this.intPt;
 72 };

 75 /**
 76  * @param {Coordinate[]}
 77  *          pts
 78  * @return {Coordinate}
 79  * @private
 80  */
 81 jsts.algorithm.CentralEndpointIntersector.average = function(pts) {
 82   var avg = new jsts.geom.Coordinate();
 83   var i, n = pts.length;
 84   for (i = 0; i < n; i++) {
 85     avg.x += pts[i].x;
 86     avg.y += pts[i].y;
 87   }
 88   if (n > 0) {
 89     avg.x /= n;
 90     avg.y /= n;
 91   }
 92   return avg;
 93 };

 96 /**
 97  * Determines a point closest to the given point.
 98  *
 99  * @param {Coordinate}
100  *          p the point to compare against.
101  * @param {Coordinate[]}
102  *          pts
103  * @return {Coordinate} the point closest to the input point p.
104  * @private
105  */
106 jsts.algorithm.CentralEndpointIntersector.prototype.findNearestPoint = function(
107     p, pts) {
108   var minDist = Number.MAX_VALUE;
109   var i, result = null, dist;
110   for (i = 0; i < pts.length; i++) {
111     dist = p.distance(pts[i]);
112     if (dist < minDist) {
113       minDist = dist;
114       result = pts[i];
115     }
116   }
117   return result;
118 };
