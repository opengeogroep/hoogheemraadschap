 11 /**
 12  * Defines a rectangular region of the 2D coordinate plane. It is often used to
 13  * represent the bounding box of a {@link Geometry}, e.g. the minimum and
 14  * maximum x and y values of the {@link Coordinate}s.
 15  * <p>
 16  * Note that Envelopes support infinite or half-infinite regions, by using the
 17  * values of <code>Double.POSITIVE_INFINITY</code> and
 18  * <code>Double.NEGATIVE_INFINITY</code>.
 19  * <p>
 20  * When Envelope objects are created or initialized, the supplies extent values
 21  * are automatically sorted into the correct order.
 22  */
 23 
 24

 27  * Creates an <code>Envelope</code> for a region defined by maximum and
 28  * minimum values.
 29  *
 30  * @constructor
 31  */

 33   jsts.geom.Envelope.prototype.init.apply(this, arguments);
 34 };

 69 /**
 70  * Creates an <code>Envelope</code> for a region defined by maximum and
 71  * minimum values.
 72  *
 73  * Will call appropriate init* method depending on arguments.
 74  */
 75 jsts.geom.Envelope.prototype.init = function() {
 76   if (typeof arguments[0] === 'number' && arguments.length === 4) {
 77     this.initFromValues(arguments[0], arguments[1], arguments[2], arguments[3]);
 78   } else if (arguments[0] instanceof jsts.geom.Coordinate &&
 79       arguments.length === 1) {
 80     this.initFromCoordinate(arguments[0]);
 81   } else if (arguments[0] instanceof jsts.geom.Coordinate &&
 82       arguments.length === 2) {
 83     this.initFromCoordinates(arguments[0], arguments[1]);
 84   } else if (arguments[0] instanceof jsts.geom.Envelope &&
 85       arguments.length === 1) {
 86     this.initFromEnvelope(arguments[0]);
 87   }
 88 };

 91 /**
 92  * Initialize an <code>Envelope</code> for a region defined by maximum and
 93  * minimum values.
 94  *
 95  * @param {number}
 96  *          x1 the first x-value.
 97  * @param {number}
 98  *          x2 the second x-value.
 99  * @param {number}
100  *          y1 the first y-value.
101  * @param {number}
102  *          y2 the second y-value.
103  */
104 jsts.geom.Envelope.prototype.initFromValues = function(x1, x2, y1, y2) {
105   if (x1 < x2) {
106     this.minx = x1;
107     this.maxx = x2;
108   } else {
109     this.minx = x2;
110     this.maxx = x1;
111   }
112   if (y1 < y2) {
113     this.miny = y1;
114     this.maxy = y2;
115   } else {
116     this.miny = y2;
117     this.maxy = y1;
118   }
119 };

294 /**
295  * Enlarges this <code>Envelope</code> so that it contains the given point.
296  * Has no effect if the point is already on or within the envelope.
297  *
298  * @param {number}
299  *          x the value to lower the minimum x to or to raise the maximum x to.
300  * @param {number}
301  *          y the value to lower the minimum y to or to raise the maximum y to.
302  */
303 jsts.geom.Envelope.prototype.expandToIncludeValues = function(x, y) {
304   if (this.isNull()) {
305     this.minx = x;
306     this.maxx = x;
307     this.miny = y;
308     this.maxy = y;
309   } else {
310     if (x < this.minx) {
311       this.minx = x;
312     }
313     if (x > this.maxx) {
314       this.maxx = x;
315     }
316     if (y < this.miny) {
317       this.miny = y;
318     }
319     if (y > this.maxy) {
320       this.maxy = y;
321     }
322   }
323 };

326 /**
327  * Enlarges this <code>Envelope</code> so that it contains the
328  * <code>other</code> Envelope. Has no effect if <code>other</code> is
329  * wholly on or within the envelope.
330  *
331  * @param {jsts.geom.Envelope}
332  *          other the <code>Envelope</code> to expand to include.
333  */
334 jsts.geom.Envelope.prototype.expandToIncludeEnvelope = function(other) {
335   if (other.isNull()) {
336     return;
337   }
338   if (this.isNull()) {
339     this.minx = other.getMinX();
340     this.maxx = other.getMaxX();
341     this.miny = other.getMinY();
342     this.maxy = other.getMaxY();
343   } else {
344     if (other.minx < this.minx) {
345       this.minx = other.minx;
346     }
347     if (other.maxx > this.maxx) {
348       this.maxx = other.maxx;
349     }
350     if (other.miny < this.miny) {
351       this.miny = other.miny;
352     }
353     if (other.maxy > this.maxy) {
354       this.maxy = other.maxy;
355     }
356   }
357 };

374 /**
375  * Expands this envelope by a given distance in all directions. Both positive
376  * and negative distances are supported.
377  *
378  * @param {number}
379  *          distance the distance to expand the envelope.
380  */
381 jsts.geom.Envelope.prototype.expandByDistance = function(distance) {
382   this.expandByDistances(distance, distance);
383 };

386 /**
387  * Expands this envelope by a given distance in all directions. Both positive
388  * and negative distances are supported.
389  *
390  * @param {number}
391  *          deltaX the distance to expand the envelope along the the X axis.
392  * @param {number}
393  *          deltaY the distance to expand the envelope along the the Y axis.
394  */
395 jsts.geom.Envelope.prototype.expandByDistances = function(deltaX, deltaY) {
396   if (this.isNull()) {
397     return;
398   }
399 
400   this.minx -= deltaX;
401   this.maxx += deltaX;
402   this.miny -= deltaY;
403   this.maxy += deltaY;
404 
405   // check for envelope disappearing
406   if (this.minx > this.maxx || this.miny > this.maxy) {
407     this.setToNull();
408   }
409 };

412 /**
413  * Translates this envelope by given amounts in the X and Y direction.
414  *
415  * @param {number}
416  *          transX the amount to translate along the X axis.
417  * @param {number}
418  *          transY the amount to translate along the Y axis.

420 jsts.geom.Envelope.prototype.translate = function(transX, transY) {
421   if (this.isNull()) {

424   this.init(this.minx + transX, this.maxx + transX, this.miny + transY,

429 /**
430  * Computes the coordinate of the centre of this envelope (as long as it is
431  * non-null
432  *
433  * @return {jsts.geom.Coordinate} the centre coordinate of this envelope <code>null</code>
434  *         if the envelope is null.
435  */
436 jsts.geom.Envelope.prototype.centre = function() {
437   if (this.isNull()) {
438     return null;
439   }
440   return new jsts.geom.Coordinate((this.minx + this.maxx) / 2.0,
441       (this.miny + this.maxy) / 2.0);
442 };

445 /**
446  * Computes the intersection of two {@link Envelopes}
447  *
448  * @param {jsts.geom.Envelope}
449  *          env the envelope to intersect with.
450  * @return {jsts.geom.Envelope} a new Envelope representing the intersection of
451  *         the envelopes (this will be the null envelope if either argument is
452  *         null, or they do not intersect.
453  */
454 jsts.geom.Envelope.prototype.intersection = function(env) {
455   if (this.isNull() || env.isNull() || !this.intersects(env)) {
456     return new jsts.geom.Envelope();
457   }
458 
459   var intMinX = this.minx > env.minx ? this.minx : env.minx;
460   var intMinY = this.miny > env.miny ? this.miny : env.miny;
461   var intMaxX = this.maxx < env.maxx ? this.maxx : env.maxx;
462   var intMaxY = this.maxy < env.maxy ? this.maxy : env.maxy;
463 
464   return new jsts.geom.Envelope(intMinX, intMaxX, intMinY, intMaxY);
465 };

468 /**
469  * Check if the region defined by input overlaps (intersects) the region of this
470  * <code>Envelope</code>.
471  *
472  * Will call appropriate intersects* depending on arguments.
473  *
474  * @return {boolean} <code>true</code> if an overlap is found.
475  */
476 jsts.geom.Envelope.prototype.intersects = function() {
477   if (arguments[0] instanceof jsts.geom.Envelope) {
478     return this.intersectsEnvelope(arguments[0]);
479   } else if (arguments[0] instanceof jsts.geom.Coordinate) {
480     return this.intersectsCoordinate(arguments[0]);
481   } else {
482     return this.intersectsValues(arguments[0], arguments[1]);
483   }
484 };

487 /**
488  * Check if the region defined by <code>other</code> overlaps (intersects) the
489  * region of this <code>Envelope</code>.
490  *
491  * @param {jsts.geom.Envelope}
492  *          other the <code>Envelope</code> which this <code>Envelope</code>
493  *          is being checked for overlapping.
494  * @return {boolean} <code>true</code> if the <code>Envelope</code>s
495  *         overlap.
496  */
497 jsts.geom.Envelope.prototype.intersectsEnvelope = function(other) {
498   if (this.isNull() || other.isNull()) {
499     return false;
500   }
501 
502   var result = !(other.minx > this.maxx || other.maxx < this.minx ||
503       other.miny > this.maxy || other.maxy < this.miny);
504   return result;
505 };

689 /**
690  * Computes the distance between this and another <code>Envelope</code>.
691  *
692  * @param {jsts.geom.Envelope}
693  *          env The <code>Envelope</code> to test this <code>Envelope</code>
694  *          against.
695  * @return {number} The distance between overlapping Envelopes is 0. Otherwise,
696  *         the distance is the Euclidean distance between the closest points.
697  */
698 jsts.geom.Envelope.prototype.distance = function(env) {
699   if (this.intersects(env)) {
700     return 0;
701   }
702   var dx = 0.0;
703   if (this.maxx < env.minx) {
704     dx = env.minx - this.maxx;
705   }
706   if (this.minx > env.maxx) {
707     dx = this.minx - env.maxx;
708   }
709 
710   var dy = 0.0;
711   if (this.maxy < env.miny) {
712     dy = env.miny - this.maxy;
713   }
714   if (this.miny > env.maxy) {
715     dy = this.miny - env.maxy;
716   }
717 
718   // if either is zero, the envelopes overlap either vertically or horizontally
719   if (dx === 0.0) {
720     return dy;
721   }
722   if (dy === 0.0) {
723     return dx;
724   }
725   return Math.sqrt(dx * dx + dy * dy);
726 };

731  *          other the <code>Envelope</code> to check against.
732  * @return {boolean} true if envelopes are equal.

734 jsts.geom.Envelope.prototype.equals = function(other) {
735   if (this.isNull()) {

737   }
738   return this.maxx === other.maxx && this.maxy === other.maxy &&
739       this.minx === other.minx && this.miny === other.miny;

752 /**
753  * Test the point q to see whether it intersects the Envelope defined by p1-p2
754  *
755  * NOTE: calls intersectsEnvelope if four arguments are given to simulate
756  * overloaded function
757  *
758  * @param {jsts.geom.Coordinate}
759  *          p1 one extremal point of the envelope.
760  * @param {jsts.geom.Coordinate}
761  *          p2 another extremal point of the envelope.
762  * @param {jsts.geom.Coordinate}
763  *          q the point to test for intersection.
764  * @return {boolean} <code>true</code> if q intersects the envelope p1-p2.
765  */
766 jsts.geom.Envelope.intersects = function(p1, p2, q) {
767   if (arguments.length === 4) {
768     return jsts.geom.Envelope.intersectsEnvelope(arguments[0], arguments[1],
769         arguments[2], arguments[3]);
770   }
771 
772   var xc1 = p1.x < p2.x ? p1.x : p2.x;
773   var xc2 = p1.x > p2.x ? p1.x : p2.x;
774   var yc1 = p1.y < p2.y ? p1.y : p2.y;
775   var yc2 = p1.y > p2.y ? p1.y : p2.y;
776 
777   if (((q.x >= xc1) && (q.x <= xc2)) && ((q.y >= yc1) && (q.y <= yc2))) {
778     return true;
779   }
780   return false;
781 };

784 /**
785  * Test the envelope defined by p1-p2 for intersection with the envelope defined
786  * by q1-q2
787  *
788  * @param {jsts.geom.Coordinate}
789  *          p1 one extremal point of the envelope P.
790  * @param {jsts.geom.Coordinate}
791  *          p2 another extremal point of the envelope P.
792  * @param {jsts.geom.Coordinate}
793  *          q1 one extremal point of the envelope Q.
794  * @param {jsts.geom.Coordinate}
795  *          q2 another extremal point of the envelope Q.
796  * @return {boolean} <code>true</code> if Q intersects P.
797  */
798 jsts.geom.Envelope.intersectsEnvelope = function(p1, p2, q1, q2) {
799   var minq = Math.min(q1.x, q2.x);
800   var maxq = Math.max(q1.x, q2.x);
801   var minp = Math.min(p1.x, p2.x);
802   var maxp = Math.max(p1.x, p2.x);
803 
804   if (minp > maxq) {
805     return false;
806   }
807   if (maxp < minq) {
808     return false;
809   }
810 
811   minq = Math.min(q1.y, q2.y);
812   maxq = Math.max(q1.y, q2.y);
813   minp = Math.min(p1.y, p2.y);
814   maxp = Math.max(p1.y, p2.y);
815 
816   if (minp > maxq) {
817     return false;
818   }
819   if (maxp < minq) {
820     return false;
821   }
822   return true;
823 };
