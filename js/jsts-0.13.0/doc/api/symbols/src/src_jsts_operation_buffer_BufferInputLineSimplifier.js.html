  9 /**
 10  * Simplifies a buffer input line to remove concavities with shallow depth.
 11  * <p>
 12  * The most important benefit of doing this is to reduce the number of points
 13  * and the complexity of shape which will be buffered. It also reduces the risk
 14  * of gores created by the quantized fillet arcs (although this issue should be
 15  * eliminated in any case by the offset curve generation logic).
 16  * <p>
 17  * A key aspect of the simplification is that it affects inside (concave or
 18  * inward) corners only. Convex (outward) corners are preserved, since they are
 19  * required to ensure that the generated buffer curve lies at the correct
 20  * distance from the input geometry.
 21  * <p>
 22  * Another important heuristic used is that the end segments of the input are
 23  * never simplified. This ensures that the client buffer code is able to
 24  * generate end caps faithfully.
 25  * <p>
 26  * No attempt is made to avoid self-intersections in the output. This is
 27  * acceptable for use for generating a buffer offset curve, since the buffer
 28  * algorithm is insensitive to invalid polygonal geometry. However, this means
 29  * that this algorithm cannot be used as a general-purpose polygon
 30  * simplification technique.
 31  *
 32  * @constructor
 33  */
 34 jsts.operation.buffer.BufferInputLineSimplifier = function(inputLine) {
 35   this.inputLine = inputLine;
 36 };
 37 
 38 
 39 /**
 40  * Simplify the input coordinate list. If the distance tolerance is positive,
 41  * concavities on the LEFT side of the line are simplified. If the supplied
 42  * distance tolerance is negative, concavities on the RIGHT side of the line are
 43  * simplified.
 44  *
 45  * @param inputLine
 46  *          the coordinate list to simplify.
 47  * @param distanceTol
 48  *          simplification distance tolerance to use.
 49  * @return the simplified coordinate list.
 50  */
 51 jsts.operation.buffer.BufferInputLineSimplifier.simplify = function(inputLine,
 52     distanceTol) {
 53   var simp = new jsts.operation.buffer.BufferInputLineSimplifier(inputLine);
 54   return simp.simplify(distanceTol);
 55 };

 98 jsts.operation.buffer.BufferInputLineSimplifier.prototype.angleOrientation = jsts.algorithm.CGAlgorithms.COUNTERCLOCKWISE;

101 /**
102  * Simplify the input coordinate list. If the distance tolerance is positive,
103  * concavities on the LEFT side of the line are simplified. If the supplied
104  * distance tolerance is negative, concavities on the RIGHT side of the line are
105  * simplified.
106  *
107  * @param distanceTol
108  *          simplification distance tolerance to use.
109  * @return the simplified coordinate list.
110  */
111 jsts.operation.buffer.BufferInputLineSimplifier.prototype.simplify = function(
112     distanceTol) {
113   this.distanceTol = Math.abs(distanceTol);
114   if (distanceTol < 0)
115     this.angleOrientation = jsts.algorithm.CGAlgorithms.CLOCKWISE;
116 
117   // rely on fact that boolean array is filled with false value
118   // TODO: need to check if this holds true in js
119   this.isDeleted = [];
120   this.isDeleted.length = this.inputLine.length;
121 
122   var isChanged = false;
123   do {
124     isChanged = this.deleteShallowConcavities();
125   } while (isChanged);
126 
127   return this.collapseLine();
128 };
129 
130 
131 /**
132  * Uses a sliding window containing 3 vertices to detect shallow angles in which
133  * the middle vertex can be deleted, since it does not affect the shape of the
134  * resulting buffer in a significant way.
135  *
136  * @return
137  * @private
138  */
139 jsts.operation.buffer.BufferInputLineSimplifier.prototype.deleteShallowConcavities = function() {
140   /**
141    * Do not simplify end line segments of the line string. This ensures that end
142    * caps are generated consistently.
143    */
144   var index = 1;
145   var maxIndex = this.inputLine.length - 1;
146 
147   var midIndex = this.findNextNonDeletedIndex(index);
148   var lastIndex = this.findNextNonDeletedIndex(midIndex);
149 
150   var isChanged = false;
151   while (lastIndex < this.inputLine.length) {
152     // test triple for shallow concavity
153     var isMiddleVertexDeleted = false;
154     if (this.isDeletable(index, midIndex, lastIndex, this.distanceTol)) {
155       this.isDeleted[midIndex] = jsts.operation.buffer.BufferInputLineSimplifier.DELETE;
156       isMiddleVertexDeleted = true;
157       isChanged = true;
158     }
159     // move simplification window forward
160     if (isMiddleVertexDeleted)
161       index = lastIndex;
162     else
163       index = midIndex;
164 
165     midIndex = this.findNextNonDeletedIndex(index);
166     lastIndex = this.findNextNonDeletedIndex(midIndex);
167   }
168   return isChanged;
169 };
170 
171 
172 /**
173  * Finds the next non-deleted index, or the end of the point array if none
174  *
175  * @param index
176  * @return the next non-deleted index, if any.
177  * @return inputLine.length if there are no more non-deleted indices.
178  * @private
179  */
180 jsts.operation.buffer.BufferInputLineSimplifier.prototype.findNextNonDeletedIndex = function(
181     index) {
182   var next = index + 1;
183   while (next < this.inputLine.length &&
184       this.isDeleted[next] === jsts.operation.buffer.BufferInputLineSimplifier.DELETE)
185     next++;
186   return next;
187 };

190 /**
191  * @private
192  */
193 jsts.operation.buffer.BufferInputLineSimplifier.prototype.collapseLine = function() {
194   var coordList = [];
195   for (var i = 0; i < this.inputLine.length; i++) {
196     if (this.isDeleted[i] !== jsts.operation.buffer.BufferInputLineSimplifier.DELETE)
197       coordList.push(this.inputLine[i]);
198   }
199 
200   return coordList;
201 };

204 /**
205  * @private
206  */
207 jsts.operation.buffer.BufferInputLineSimplifier.prototype.isDeletable = function(
208     i0, i1, i2, distanceTol) {
209   var p0 = this.inputLine[i0];
210   var p1 = this.inputLine[i1];
211   var p2 = this.inputLine[i2];
212 
213   if (!this.isConcave(p0, p1, p2))
214     return false;
215   if (!this.isShallow(p0, p1, p2, distanceTol))
216     return false;
217 
218   return this.isShallowSampled(p0, p1, i0, i2, distanceTol);
219 };
220 
221 
222 /**
223  * @private
224  */
225 jsts.operation.buffer.BufferInputLineSimplifier.prototype.isShallowConcavity = function(
226     p0, p1, p2, distanceTol) {
227   var orientation = jsts.algorithm.CGAlgorithms.computeOrientation(p0, p1, p2);
228   var isAngleToSimplify = (orientation === this.angleOrientation);
229   if (!isAngleToSimplify)
230     return false;
231 
232   var dist = jsts.algorithm.CGAlgorithms.distancePointLine(p1, p0, p2);
233   return dist < distanceTol;
234 };

237 /**
238  * @private
239  */
240 jsts.operation.buffer.BufferInputLineSimplifier.NUM_PTS_TO_CHECK = 10;
241 
242 
243 /**
244  * Checks for shallowness over a sample of points in the given section. This
245  * helps prevents the siplification from incrementally "skipping" over points
246  * which are in fact non-shallow.
247  *
248  * @param p0
249  *          start coordinate of section.
250  * @param p2
251  *          end coordinate of section.
252  * @param i0
253  *          start index of section.
254  * @param i2
255  *          end index of section.
256  * @param distanceTol
257  *          distance tolerance.
258  * @return
259  * @private
260  */
261 jsts.operation.buffer.BufferInputLineSimplifier.prototype.isShallowSampled = function(
262     p0, p2, i0, i2, distanceTol) {
263   // check every n'th point to see if it is within tolerance
264   var inc = (i2 - i0) /
265       jsts.operation.buffer.BufferInputLineSimplifier.NUM_PTS_TO_CHECK;
266   if (inc <= 0)
267     inc = 1;
268 
269   for (var i = i0; i < i2; i += inc) {
270     if (!this.isShallow(p0, p2, this.inputLine[i], distanceTol))
271       return false;
272   }
273   return true;
274 };

277 /**
278  * @private
279  */
280 jsts.operation.buffer.BufferInputLineSimplifier.prototype.isShallow = function(
281     p0, p1, p2, distanceTol) {
282   var dist = jsts.algorithm.CGAlgorithms.distancePointLine(p1, p0, p2);
283   return dist < distanceTol;
284 };

287 /**
288  * @private
289  */
290 jsts.operation.buffer.BufferInputLineSimplifier.prototype.isConcave = function(
291     p0, p1, p2) {
292   var orientation = jsts.algorithm.CGAlgorithms.computeOrientation(p0, p1, p2);
293   var isConcave = (orientation === this.angleOrientation);
294   return isConcave;
295 };
