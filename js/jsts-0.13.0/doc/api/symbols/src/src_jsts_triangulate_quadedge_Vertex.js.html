  8 /**
  9  * Models a site (node) in a {@link QuadEdgeSubdivision}. The sites can be
 10  * points on a lineString representing a linear site. The vertex can be
 11  * considered as a vector with a norm, length, inner product, cross product,
 12  * etc. Additionally, point relations (e.g., is a point to the left of a line,
 13  * the circle defined by this point and two others, etc.) are also defined in
 14  * this class.
 15  *
 16  * Initializes a new Vertex. Will call the correct init* -function based on
 17  * arguments
 18  *
 19  * @constructor
 20  */
 21 
 22 jsts.triangulate.quadedge.Vertex = function() {
 23   if (arguments.length === 1) {
 24     this.initFromCoordinate(arguments[0]);
 25   } else {
 26     this.initFromXY(arguments[0], arguments[1]);
 27   }
 28 };

160  *

162  *

201 /**
202  * Clasifys a vertex with respect to another vertex
203  *
204  * @param {jsts.triangulate.quadedge.Vertex}
205  *          p0 The first vertex.
206  * @param {jsts.triangulate.quadedge.Vertex}
207  *          p1 The second vertex.
208  * @return {Number} The classification.
209  */
210 jsts.triangulate.quadedge.Vertex.prototype.classify = function(p0, p1) {
211   var p2, a, b, sa;
212 
213   p2 = this;
214   a = p1.sub(p0);
215   b = p2.sub(p0);
216   sa = a.crossProduct(b);
217 
218   if (sa > 0.0) {
219     return jsts.triangulate.quadedge.Vertex.LEFT;
220   }
221   if (sa < 0.0) {
222     return jsts.triangulate.quadedge.Vertex.RIGHT;
223   }
224   if ((a.getX() * b.getX() < 0.0) || (a.getY() * b.getY() < 0.0)) {
225     return jsts.triangulate.quadedge.Vertex.BEHIND;
226   }
227   if (a.magn() < b.magn()) {
228     return jsts.triangulate.quadedge.Vertex.BEYOND;
229   }
230   if (p0.equals(p2)) {
231     return jsts.triangulate.quadedge.Vertex.ORIGIN;
232   }
233   if (p1.equals(p2)) {
234     return jsts.triangulate.quadedge.Vertex.DESTINATION;
235   }
236   return jsts.triangulate.quadedge.Vertex.BETWEEN;
237 };

245  * @return {Number} The magnitude of u X v.

247 jsts.triangulate.quadedge.Vertex.prototype.crossProduct = function(v) {
248   return ((this.p.x * v.getY()) - (this.p.y * v.getX()));

340 /**
341  * Tests whether the triangle formed by this vertex and two other vertices is in
342  * CCW orientation.
343  *
344  * @param {jsts.triangulate.quadedge.Vertex}
345  *          b a vertex.
346  * @param {jsts.triangulate.quadedge.Vertex}
347  *          c a vertex.
348  * @return {Boolean} true if the triangle is oriented CCW.
349  */
350 jsts.triangulate.quadedge.Vertex.prototype.isCCW = function(b, c) {
351   // is equal to the signed area of the triangle
352   return ((b.p.x - this.p.x) * (c.p.y - this.p.y) - (b.p.y - this.p.y) *
353       (c.p.x - this.p.x) > 0);
354 };

381 /**
382  * Returns the perpendicular bisector of the line between the input vertices
383  *
384  * @param {jsts.triangulate.quadedge.Vertex}
385  *          a A vertex.
386  * @param {jsts.triangulate.quadedge.Vertex}
387  *          b A vertex.
388  * @return {jsts.algorithm.HCoordinate} The bisector.
389  */
390 jsts.triangulate.quadedge.Vertex.prototype.bisector = function(a, b) {
391   var dx, dy, l1, l2;
392 
393   dx = b.getX() - a.getX();
394   dy = b.getY() - a.getY();
395 
396   l1 = new jsts.algorithm.HCoordinate(a.getX() + (dx / 2.0), a.getY() +
397       (dy / 2.0), 1.0);
398   l1 = new jsts.algorithm.HCoordinate(a.getX() - dy + (dx / 2.0), a.getY() +
399       dx + (dy / 2.0), 1.0);
400   return new jsts.algorithm.HCoordinate(l1, l2);
401 };

418 /**
419  * Computes the value of the ratio of the circumradius to shortest edge. If
420  * smaller than some given tolerance B, the associated triangle is considered
421  * skinny.
422  *
423  * For an equal lateral triangle this value is 0.57735. The ratio is related to
424  * the minimum triangle angle theta by: circumRadius/shortestEdge =
425  * 1/(2sin(theta)).
426  *
427  * @param {jsts.triangulate.quadedge}
428  *          b second vertex of the triangle.
429  * @param {jsts.triangulate.quadedge}
430  *          c third vertex of the triangle.
431  * @return {Number} ratio of circumradius to shortest edge.
432  */
433 jsts.triangulate.quadedge.Vertex.prototype.circumRadiusRatio = function(b, c) {
434   var x, radius, edgeLength, el;
435 
436   x = this.circleCenter(b, c);
437   radius = this.distance(x, b);
438   edgeLength = this.distance(this, b);
439   el = this.distance(b, c);
440 
441   if (el < edgeLength) {
442     edgeLength = el;
443   }
444   el = this.distance(c, this);
445   if (el < edgeLength) {
446     edgeLength = el;
447   }
448   return radius / edgeLength;
449 };

461 jsts.triangulate.quadedge.Vertex.prototype.midPoint = function(a) {
462   var xm, ym;

464   ym = (this.p.y + a.getY()) / 2.0;
465

470 /**
471  * Computes the centre of the circumcircle of this vertex and two others.
472  *
473  * @param {jsts.triangulate.quadedge.Vertex}
474  *          b a vertex.
475  * @param {jsts.triangulate.quadedge.Vertex}
476  *          c a vertex.
477  * @return {jsts.triangulate.quadedge.Vertex} the Coordinate which is the
478  *         circumcircle of the 3 points.
479  */
480 jsts.triangulate.quadedge.Vertex.prototype.circleCenter = function(b, c) {
481   var a, cab, cbc, hcc, cc;
482 
483   a = new jsts.triangulate.quadedge.Vertex(this.getX(), this.getY());
484   // compute the perpendicular bisector of cord ab
485   cab = this.bisector(a, b);
486   // compute the perpendicular bisector of cord bc
487   cbc = this.bisector(b, c);
488   // compute the intersection of the bisectors (circle radii)
489   hcc = new jsts.algorithm.HCoordinate(cab, cbc);
490   cc = null;
491   try {
492     cc = new jsts.triangulate.quadedge.Vertex(hcc.getX(), hcc.getY());
493   } catch (err) {
494   }
495 
496   return cc;
497 };
