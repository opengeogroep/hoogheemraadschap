  7 (function() {
  8 
  9   /**
 10    * @requires jsts/geom/Geometry.js
 11    */
 12 
 13   var Geometry = jsts.geom.Geometry;
 14   var TreeSet = javascript.util.TreeSet;
 15   var Arrays = javascript.util.Arrays;
 16 
 17   /**
 18    * @constructor
 19    * @extends jsts.geom.Geometry
 20    */
 21   jsts.geom.GeometryCollection = function(components) {
 22     OpenLayers.Geometry.prototype.initialize.apply(this, arguments);
 23 
 24     this.geometries = [];
 25 
 26     if (!components) {
 27       return;
 28     }
 29 
 30     for (var i = 0; i < components.length; i++) {
 31       var component = components[i];
 32       if (component instanceof jsts.geom.Point) {
 33         components[i] = component.coordinate;
 34       } else if (component instanceof jsts.geom.Coordinate) {
 35         component = new jsts.geom.Point(component);
 36       }
 37       this.geometries.push(component);
 38     }
 39 
 40     this.components = [];
 41     this.addComponents(components);
 42   };
 43 
 44   jsts.geom.GeometryCollection.prototype = new OpenLayers.Geometry.Collection();
 45 
 46   for (key in Geometry.prototype) {
 47     jsts.geom.GeometryCollection.prototype[key] = Geometry.prototype[key];
 48   }
 49 
 50 
 51   /**
 52    * @return {boolean}
 53    */
 54   jsts.geom.GeometryCollection.prototype.isEmpty = function() {
 55     for (var i = 0; i < this.geometries.length; i++) {
 56       var geometry = this.geometries[i];
 57 
 58       if (!geometry.isEmpty()) {
 59         return false;
 60       }
 61     }
 62     return true;
 63   };
 64 
 65 
 66   /**
 67    * @return {Coordinate}
 68    */
 69   jsts.geom.GeometryCollection.prototype.getCoordinate = function() {
 70     if (this.isEmpty())
 71       return null;
 72 
 73     return this.geometries[0].getCoordinate();
 74   };
 75 
 76 
 77   /**
 78    * Collects all coordinates of all subgeometries into an Array.
 79    *
 80    * Note that while changes to the coordinate objects themselves may modify the
 81    * Geometries in place, the returned Array as such is only a temporary
 82    * container which is not synchronized back.
 83    *
 84    * @return {Coordinate[]} the collected coordinates.
 85    */
 86   jsts.geom.GeometryCollection.prototype.getCoordinates = function() {
 87     var coordinates = [];
 88     var k = -1;
 89     for (var i = 0; i < this.geometries.length; i++) {
 90       var geometry = this.geometries[i];
 91 
 92       var childCoordinates = geometry.getCoordinates();
 93       for (var j = 0; j < childCoordinates.length; j++) {
 94         k++;
 95         coordinates[k] = childCoordinates[j];
 96       }
 97     }
 98     return coordinates;
 99   };
100 
101 
102   /**
103    * @return {int}
104    */
105   jsts.geom.GeometryCollection.prototype.getNumGeometries = function() {
106     return this.geometries.length;
107   };
108 
109 
110   /**
111    * @param {int}
112    *          n
113    * @return {Geometry}
114    */
115   jsts.geom.GeometryCollection.prototype.getGeometryN = function(n) {
116     return this.geometries[n];
117   };
118 
119 
120   /**
121    * @param {Geometry}
122    *          other
123    * @param {double}
124    *          tolerance
125    * @return {boolean}
126    */
127   jsts.geom.GeometryCollection.prototype.equalsExact = function(other,
128       tolerance) {
129     if (!this.isEquivalentClass(other)) {
130       return false;
131     }
132     if (this.geometries.length !== other.geometries.length) {
133       return false;
134     }
135     for (var i = 0; i < this.geometries.length; i++) {
136       var geometry = this.geometries[i];
137 
138       if (!geometry.equalsExact(other.geometries[i], tolerance)) {
139         return false;
140       }
141     }
142     return true;
143   };
144 
145   jsts.geom.GeometryCollection.prototype.compareToSameClass = function(o) {
146     var theseElements = new TreeSet(Arrays.asList(this.geometries));
147     var otherElements = new TreeSet(Arrays.asList(o.geometries));
148     return this.compare(theseElements, otherElements);
149   };
150 
151   jsts.geom.GeometryCollection.prototype.apply = function(filter) {
152     if (filter instanceof jsts.geom.GeometryFilter || filter instanceof jsts.geom.GeometryComponentFilter) {
153       filter.filter(this);
154       for (var i = 0; i < this.geometries.length; i++) {
155         this.geometries[i].apply(filter);
156       }
157     } else if (filter instanceof jsts.geom.CoordinateFilter) {
158       for (var i = 0; i < this.geometries.length; i++) {
159         this.geometries[i].apply(filter);
160       }
161     }
162   };
163 
164 
165   jsts.geom.GeometryCollection.prototype.getDimension = function() {
166     var dimension = jsts.geom.Dimension.FALSE;
167     for (var i = 0; i < this.geometries.length; i++) {
168       var geometry = this.geometries[i];
169       dimension = Math.max(dimension, geometry.getDimension());
170     }
171     return dimension;
172   };
173 
174 
175   /**
176    * @protected
177    */
178   jsts.geom.GeometryCollection.prototype.computeEnvelopeInternal = function() {
179     var envelope = new jsts.geom.Envelope();
180     for (var i = 0; i < this.geometries.length; i++) {
181       var geometry = this.geometries[i];
182       envelope.expandToInclude(geometry.getEnvelopeInternal());
183     }
184     return envelope;
185   };
186 
187   /**
188    * @private
189    */
190   OpenLayers.Geometry.Collection = jsts.geom.GeometryCollection;
191 
192 })();
