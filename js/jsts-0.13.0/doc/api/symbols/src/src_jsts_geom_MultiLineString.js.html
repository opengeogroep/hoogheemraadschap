  7 (function() {
  8 
  9 
 10   /**
 11    * @requires jsts/geom/GeometryCollection.js
 12    */
 13 
 14 
 15   /**
 16    * @constructor
 17    * @extends jsts.geom.GeometryCollection
 18    */
 19   jsts.geom.MultiLineString = function() {
 20     if (arguments[0] instanceof Array) {
 21       // NOTE: need to support LinearRings as input (OL only supports
 22       // LineStrings
 23       var array = arguments[0];
 24       for (var i = 0; i < array.length; i++) {
 25         var e = array[i];
 26         if (e.CLASS_NAME === 'OpenLayers.Geometry.LinearRing') {
 27           array[i] = new jsts.geom.LineString(e.components);
 28         }
 29       }
 30     }
 31 
 32     OpenLayers.Geometry.Collection.prototype.initialize.apply(this, arguments);
 33     this.geometries = this.components;
 34   };
 35   jsts.geom.MultiLineString.prototype = OpenLayers.Geometry.MultiLineString.prototype;
 36 
 37   for (key in jsts.geom.GeometryCollection.prototype) {
 38     jsts.geom.MultiLineString.prototype[key] = jsts.geom.MultiLineString.prototype[key] ? jsts.geom.MultiLineString.prototype[key]
 39         : jsts.geom.GeometryCollection.prototype[key];
 40 
 41     // overrides
 42     jsts.geom.MultiLineString.prototype.getCentroid = jsts.geom.GeometryCollection.prototype.getCentroid;
 43   }
 44 
 45   jsts.geom.MultiLineString.prototype.getBoundary = function() {
 46     return (new jsts.operation.BoundaryOp(this)).getBoundary();
 47   };
 48 
 49 
 50   /**
 51    * @param {Geometry}
 52    *          other
 53    * @param {double}
 54    *          tolerance
 55    * @return {boolean}
 56    */
 57   jsts.geom.MultiLineString.prototype.equalsExact = function(other, tolerance) {
 58     if (!this.isEquivalentClass(other)) {
 59       return false;
 60     }
 61     return jsts.geom.GeometryCollection.prototype.equalsExact.call(this, other,
 62         tolerance);
 63   };
 64 
 65   OpenLayers.Geometry.MultiLineString = jsts.geom.MultiLineString;
 66 
 67 })();
