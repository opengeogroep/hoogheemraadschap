  9 /**
 10  * Represents an intersection point between two {@link SegmentString}s.
 11  *
 12  * @constructor
 13  */
 14 jsts.noding.SegmentNode = function(segString,  coord,  segmentIndex,  segmentOctant) {
 15   this.segString = segString;
 16   this.coord = new jsts.geom.Coordinate(coord);
 17   this.segmentIndex = segmentIndex;
 18   this.segmentOctant = segmentOctant;
 19   this._isInterior = ! coord.equals2D(segString.getCoordinate(segmentIndex));
 20 };
 21 
 22 
 23 /**
 24  * @type {NodedSegmentString}
 25  * @private
 26  */
 27 jsts.noding.SegmentNode.prototype.segString = null;
 28 jsts.noding.SegmentNode.prototype.coord = null;   // the point of intersection
 29 jsts.noding.SegmentNode.prototype.segmentIndex = null;   // the index of the containing line segment in the parent edge
 30 jsts.noding.SegmentNode.prototype.segmentOctant = null;
 31 jsts.noding.SegmentNode.prototype._isInterior = null;
 32 
 33 
 34 jsts.noding.SegmentNode.prototype.isInterior = function() { return this._isInterior; };
 35 
 36 jsts.noding.SegmentNode.prototype.isEndPoint = function(maxSegmentIndex)  {
 37   if (this.segmentIndex === 0 && ! this._isInterior) return true;
 38   if (this.segmentIndex === this.maxSegmentIndex) return true;
 39   return false;
 40 };
 41 
 42 
 43 /**
 44    * @return -1 this SegmentNode is located before the argument location.
 45    * @return 0 this SegmentNode is at the argument location.
 46    * @return 1 this SegmentNode is located after the argument location.
 47    */
 48 jsts.noding.SegmentNode.prototype.compareTo = function(obj)  {
 49   var other = obj;
 50 
 51   if (this.segmentIndex < other.segmentIndex) return -1;
 52   if (this.segmentIndex > other.segmentIndex) return 1;
 53 
 54   if (this.coord.equals2D(other.coord)) return 0;
 55 
 56   return jsts.noding.SegmentPointComparator.compare(this.segmentOctant, this.coord, other.coord);
 57 };
 58 
 59 /**
 60  * Used as hashcode in JSTS
 61  */
 62 jsts.noding.SegmentNode.prototype.toString = function() {
 63   return 'SegmentNode_' + this.coord + '_' + this.segmentIndex;
 64 };
