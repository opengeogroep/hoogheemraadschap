 20   /**
 21    * @constructor
 22    * @extends jsts.geomgraph.EdgeEnd
 23    */
 24   jsts.geomgraph.DirectedEdge = function(edge, isForward) {
 25     EdgeEnd.call(this, edge);
 26 
 27     this.depth = [0, -999, -999];
 28 
 29     this._isForward = isForward;
 30     if (isForward) {
 31       this.init(edge.getCoordinate(0), edge.getCoordinate(1));
 32     } else {
 33       var n = edge.getNumPoints() - 1;
 34       this.init(edge.getCoordinate(n), edge.getCoordinate(n - 1));
 35     }
 36     this.computeDirectedLabel();
 37 
 38   };
 39   jsts.geomgraph.DirectedEdge.prototype = new EdgeEnd();
 40   jsts.geomgraph.DirectedEdge.constructor = jsts.geomgraph.DirectedEdge;

 43   /**
 44    * Computes the factor for the change in depth when moving from one location
 45    * to another. E.g. if crossing from the INTERIOR to the EXTERIOR the depth
 46    * decreases, so the factor is -1
 47    */
 48   jsts.geomgraph.DirectedEdge.depthFactor = function(currLocation, nextLocation) {
 49     if (currLocation === Location.EXTERIOR &&
 50         nextLocation === Location.INTERIOR)
 51       return 1;
 52     else if (currLocation === Location.INTERIOR &&
 53         nextLocation === Location.EXTERIOR)
 54       return -1;
 55     return 0;
 56   };

145   jsts.geomgraph.DirectedEdge.prototype.setDepth = function(position, depthVal) {
146     if (this.depth[position] !== -999) {
147       if (this.depth[position] !== depthVal)
148         throw new jsts.error.TopologyError('assigned depths do not match', this
149             .getCoordinate());
150     }
151     this.depth[position] = depthVal;
152   };

154   jsts.geomgraph.DirectedEdge.prototype.getDepthDelta = function() {
155     var depthDelta = this.edge.getDepthDelta();
156     if (!this._isForward)
157       depthDelta = -depthDelta;
158     return depthDelta;

198   /**
199    * This edge is a line edge if
200    * <ul>
201    * <li> at least one of the labels is a line label
202    * <li> any labels which are not line labels have all Locations = EXTERIOR
203    * </ul>
204    */
205   jsts.geomgraph.DirectedEdge.prototype.isLineEdge = function() {
206     var isLine = this.label.isLine(0) || this.label.isLine(1);
207     var isExteriorIfArea0 = !this.label.isArea(0) ||
208         this.label.allPositionsEqual(0, Location.EXTERIOR);
209     var isExteriorIfArea1 = !this.label.isArea(1) ||
210         this.label.allPositionsEqual(1, Location.EXTERIOR);
211 
212     return isLine && isExteriorIfArea0 && isExteriorIfArea1;
213   };
214   /**
215    * This is an interior Area edge if
216    * <ul>
217    * <li> its label is an Area label for both Geometries
218    * <li> and for each Geometry both sides are in the interior.
219    * </ul>
220    *
221    * @return true if this is an interior Area edge.
222    */
223   jsts.geomgraph.DirectedEdge.prototype.isInteriorAreaEdge = function() {
224     var isInteriorAreaEdge = true;
225     for (var i = 0; i < 2; i++) {
226       if (!(this.label.isArea(i) &&
227           this.label.getLocation(i, Position.LEFT) === Location.INTERIOR && this.label
228           .getLocation(i, Position.RIGHT) === Location.INTERIOR)) {
229         isInteriorAreaEdge = false;
230       }
231     }
232     return isInteriorAreaEdge;
233   };

235   /**
236    * Compute the label in the appropriate orientation for this DirEdge
237    *
238    * @private
239    */
240   jsts.geomgraph.DirectedEdge.prototype.computeDirectedLabel = function() {
241     this.label = new jsts.geomgraph.Label(this.edge.getLabel());
242     if (!this._isForward)
243       this.label.flip();
244   };

246   /**
247    * Set both edge depths. One depth for a given side is provided. The other is
248    * computed depending on the Location transition and the depthDelta of the
249    * edge.
250    */
251   jsts.geomgraph.DirectedEdge.prototype.setEdgeDepths = function(position, depth) {
252     // get the depth transition delta from R to L for this directed Edge
253     var depthDelta = this.getEdge().getDepthDelta();
254     if (!this._isForward)
255       depthDelta = -depthDelta;
256 
257     // if moving from L to R instead of R to L must change sign of delta
258     var directionFactor = 1;
259     if (position === Position.LEFT)
260       directionFactor = -1;
261 
262     var oppositePos = Position.opposite(position);
263     var delta = depthDelta * directionFactor;
264     var oppositeDepth = depth + delta;
265     this.setDepth(position, depth);
266     this.setDepth(oppositePos, oppositeDepth);
267   };
