  9 /**
 10  * Supplies a set of utility methods for building Geometry objects from lists
 11  * of Coordinates.
 12  *
 13  * Note that the factory constructor methods do <b>not</b> change the input
 14  * coordinates in any way.
 15  *
 16  * In particular, they are not rounded to the supplied <tt>PrecisionModel</tt>.
 17  * It is assumed that input Coordinates meet the given precision.
 18  *
 19  * Constructs a GeometryFactory that generates Geometries having a floating
 20  * PrecisionModel and a spatial-reference ID of 0.
 21  *
 22  * @constructor
 23  */
 24 jsts.geom.GeometryFactory = function(precisionModel) {
 25   this.precisionModel = precisionModel;
 26 };
 27 
 28 jsts.geom.GeometryFactory.precisionModel = null;
 29 
 30 
 31 /**
 32  * Creates a Point using the given Coordinate; a null Coordinate will create an
 33  * empty Geometry.
 34  *
 35  * @param {Coordinate}
 36  *          coordinate Coordinate to base this Point on.
 37  * @return {Point} A new Point.
 38  */
 39 jsts.geom.GeometryFactory.prototype.createPoint = function(coordinate) {
 40   var point = null;
 41 
 42   if (coordinate === null) {
 43     point = new jsts.geom.Point();
 44     point.setPrecisionModel(this.precisionModel);
 45   } else {
 46     point = new jsts.geom.Point(coordinate);
 47   }
 48 
 49   point.setPrecisionModel(this.precisionModel);
 50 
 51   return point;
 52 };
 53

 55 /**
 56  * Creates a LineString using the given Coordinates; a null or empty array will
 57  * create an empty LineString. Consecutive points must not be equal.
 58  *
 59  * @param {Coordinate[]}
 60  *          coordinates an array without null elements, or an empty array, or
 61  *          null.
 62  * @return {LineString} A new LineString.
 63  */
 64 jsts.geom.GeometryFactory.prototype.createLineString = function(coordinates) {
 65   var lineString = new jsts.geom.LineString(coordinates);
 66   lineString.setPrecisionModel(this.precisionModel);
 67 
 68   return lineString;
 69 };

 89 /**
 90  * Constructs a <code>Polygon</code> with the given exterior boundary and
 91  * interior boundaries.
 92  *
 93  * @param {LinearRing}
 94  *          shell the outer boundary of the new <code>Polygon</code>, or
 95  *          <code>null</code> or an empty <code>LinearRing</code> if the
 96  *          empty geometry is to be created.
 97  * @param {LinearRing[]}
 98  *          holes the inner boundaries of the new <code>Polygon</code>, or
 99  *          <code>null</code> or empty <code>LinearRing</code> s if the
100  *          empty geometry is to be created.
101  * @return {Polygon} A new Polygon.
102  */
103 jsts.geom.GeometryFactory.prototype.createPolygon = function(shell, holes) {
104   if (shell === null || shell === undefined) {
105     return new jsts.geom.Polygon();
106   }
107 
108   var rings = [shell];
109 
110   if (holes !== undefined) {
111     rings = rings.concat(holes);
112   }
113 
114   var polygon = new jsts.geom.Polygon(rings);
115 
116   polygon.setPrecisionModel(this.precisionModel);
117 
118   return polygon;
119 };

122 jsts.geom.GeometryFactory.prototype.createMultiPoint = function(coordinates) {
123   if (coordinates === undefined || coordinates === null) {
124     return new jsts.geom.MultiPoint();
125   }
126 
127   if (coordinates[0] instanceof jsts.geom.Point) {
128     var temp = [];
129     coordinates.forEach(function(point) {
130       temp.push(point.coordinate);
131     });
132     coordinates = temp;
133   }
134 
135   return new jsts.geom.MultiPoint(coordinates);
136 };

138 jsts.geom.GeometryFactory.prototype.createMultiLineString = function(lineStrings) {
139   if (lineStrings === undefined || lineStrings === null) {
140     return new jsts.geom.MultiLineString();
141   }
142 
143   return new jsts.geom.MultiLineString(lineStrings);
144 };

155 /**
156  *  Build an appropriate <code>Geometry</code>, <code>MultiGeometry</code>, or
157  *  <code>GeometryCollection</code> to contain the <code>Geometry</code>s in
158  *  it.
159  * For example:<br>
160  *
161  *  <ul>
162  *    <li> If <code>geomList</code> contains a single <code>Polygon</code>,
163  *    the <code>Polygon</code> is returned.
164  *    <li> If <code>geomList</code> contains several <code>Polygon</code>s, a
165  *    <code>MultiPolygon</code> is returned.
166  *    <li> If <code>geomList</code> contains some <code>Polygon</code>s and
167  *    some <code>LineString</code>s, a <code>GeometryCollection</code> is
168  *    returned.
169  *    <li> If <code>geomList</code> is empty, an empty <code>GeometryCollection</code>
170  *    is returned
171  *  </ul>
172  *
173  * Note that this method does not "flatten" Geometries in the input, and hence if
174  * any MultiGeometries are contained in the input a GeometryCollection containing
175  * them will be returned.
176  *
177  *@param  geomList  the <code>Geometry</code>s to combine
178  *@return {Geometry}          a <code>Geometry</code> of the "smallest", "most
179  *      type-specific" class that can contain the elements of <code>geomList</code>
180  *      .
181  */
182 jsts.geom.GeometryFactory.prototype.buildGeometry = function(geomList) {
183 
184   /**
185    * Determine some facts about the geometries in the list
186    */
187   var geomClass = null;
188   var isHeterogeneous = false;
189   var hasGeometryCollection = false;
190   for (var i = geomList.iterator(); i.hasNext(); ) {
191     var geom = i.next();
192     var partClass = geom.constructor;
193     if (geomClass === null) {
194       geomClass = partClass;
195     }
196     if (partClass !== geomClass) {
197       isHeterogeneous = true;
198     }
199     if (geom instanceof jsts.geom.GeometryCollection)
200       hasGeometryCollection = true;
201   }
202 
203   /**
204    * Now construct an appropriate geometry to return
205    */
206   // for the empty geometry, return an empty GeometryCollection
207   if (geomClass == null) {
208     return this.createGeometryCollection(null);
209   }
210   if (isHeterogeneous || hasGeometryCollection) {
211     return this.createGeometryCollection(geomList.toArray());
212   }
213   // at this point we know the collection is hetereogenous.
214   // Determine the type of the result from the first Geometry in the list
215   // this should always return a geometry, since otherwise an empty collection would have already been returned
216   var geom0 = geomList.get(0);
217   var isCollection = geomList.size() > 1;
218   if (isCollection) {
219     if (geom0 instanceof jsts.geom.Polygon) {
220       return this.createMultiPolygon(geomList.toArray());
221     }
222     else if (geom0 instanceof jsts.geom.LineString) {
223       return this.createMultiLineString(geomList.toArray());
224     }
225     else if (geom0 instanceof jsts.geom.Point) {
226       return this.createMultiPoint(geomList.toArray());
227     }
228     jsts.util.Assert.shouldNeverReachHere('Unhandled class: ' + geom0);
229   }
230   return geom0;
231 };
232 
233 jsts.geom.GeometryFactory.prototype.createGeometryCollection = function(geometries) {
234   return new jsts.geom.GeometryCollection(geometries);
235 };
