 15 /**
 16  * @constructor
 17  * @extends jsts.geom.Geometry
 18  */
 19 jsts.geom.Point = function(coordinate) {
 20   if (coordinate === undefined)
 21     return;
 22 
 23   this.coordinate = coordinate;
 24 };
 25 
 26 jsts.geom.Point.prototype = new jsts.geom.Geometry();
 27 jsts.geom.Point.constructor = jsts.geom.Point;
 28 
 29 
 30 jsts.geom.Point.prototype.coordinate = null;
 31

 33 /**
 34  * @return {number} x-axis value of this Point.

 53 /**
 54  * @return {Coordinate[]} this Point as coordinate array.
 55  */
 56 jsts.geom.Point.prototype.getCoordinates = function() {
 57   return this.isEmpty() ? [] : [this.coordinate];
 58 };
 59 
 60 jsts.geom.Point.prototype.isEmpty = function() {
 61   return this.coordinate === null;
 62 };
 63 
 64 jsts.geom.Point.prototype.equalsExact = function(other, tolerance) {
 65   if (!this.isEquivalentClass(other)) {
 66     return false;
 67   }
 68   if (this.isEmpty() && other.isEmpty()) {
 69     return true;
 70   }
 71   return this.equal(other.getCoordinate(), this.getCoordinate(), tolerance);
 72 };

 75 /**
 76  * @return {number} number of coordinates (0 or 1).
 77  */
 78 jsts.geom.Point.prototype.getNumPoints = function() {
 79   return this.isEmpty() ? 0 : 1;

 91 /**
 92  * Gets the boundary of this geometry. Zero-dimensional geometries have no
 93  * boundary by definition, so an empty GeometryCollection is returned.
 94  *
 95  * @return {GeometryCollection} an empty GeometryCollection.
 96  * @see Geometry#getBoundary
 97  */
 98 jsts.geom.Point.prototype.getBoundary = function() {
 99   return new jsts.geom.GeometryCollection(null);
100 };

103 /**
104  * @return {Envelope} Envelope of this point.
105  */
106 jsts.geom.Point.prototype.computeEnvelopeInternal = function() {
107   if (this.isEmpty()) {
108     return new jsts.geom.Envelope();
109   }
110   return new jsts.geom.Envelope(this);
111 };
112 
113 jsts.geom.Point.prototype.apply = function(filter) {
114   if (filter instanceof jsts.geom.GeometryFilter || filter instanceof jsts.geom.GeometryComponentFilter) {
115     filter.filter(this);
116   } else if (filter instanceof jsts.geom.CoordinateFilter) {
117     if (this.isEmpty()) { return; }
118     filter.filter(this.getCoordinate());
119   }
120 
121 };

123 jsts.geom.Point.prototype.clone = function() {
124   return new jsts.geom.Point(this.coordinate.clone());
125 };

152 /**
153  * A Point is valid iff:
154  * <ul>
155  * <li>the coordinate which defines it is a valid coordinate (i.e does not have
156  * an NaN X or Y ordinate)
157  * </ul>
158  *
159  * @return {boolean} true iff the Point is valid.
160  */
161 jsts.geom.Point.prototype.isValid = function() {
162   if (!jsts.operation.valid.IsValidOp.isValid(this.getCoordinate())) {
163     return false;
164   }
165   return true;
166 };

176 jsts.geom.Point.prototype.compareToSameClass = function(other) {
