  7 (function() {
  8 
  9   /**
 10    * @requires jsts/geom/Location.js
 11    * @requires jsts/geomgraph/Position.js
 12    */
 13 
 14   var Location = jsts.geom.Location;
 15   var Position = jsts.geomgraph.Position;
 16 
 17   /**
 18    * A Depth object records the topological depth of the sides of an Edge for up
 19    * to two Geometries.
 20    *
 21    * @constructor
 22    */
 23   jsts.geomgraph.Depth = function() {
 24     // initialize depth array to a sentinel value
 25     this.depth = [[], []];
 26     for (var i = 0; i < 2; i++) {
 27       for (var j = 0; j < 3; j++) {
 28         this.depth[i][j] = jsts.geomgraph.Depth.NULL_VALUE;
 29       }
 30     }
 31   };
 32 
 33   jsts.geomgraph.Depth.NULL_VALUE = -1;
 34 
 35   jsts.geomgraph.Depth.depthAtLocation = function(location) {
 36     if (location == Location.EXTERIOR)
 37       return 0;
 38     if (location == Location.INTERIOR)
 39       return 1;
 40     return jsts.geomgraph.Depth.NULL_VALUE;
 41   };
 42 
 43   jsts.geomgraph.Depth.prototype.depth = null;
 44 
 45 
 46   jsts.geomgraph.Depth.prototype.getDepth = function(geomIndex, posIndex) {
 47     return this.depth[geomIndex][posIndex];
 48   };
 49 
 50   jsts.geomgraph.Depth.prototype.setDepth = function(geomIndex, posIndex, depthValue) {
 51     this.depth[geomIndex][posIndex] = depthValue;
 52   };
 53 
 54   jsts.geomgraph.Depth.prototype.getLocation = function(geomIndex, posIndex) {
 55     if (this.depth[geomIndex][posIndex] <= 0)
 56       return Location.EXTERIOR;
 57     return Location.INTERIOR;
 58   };
 59 
 60   jsts.geomgraph.Depth.prototype.add = function(geomIndex, posIndex, location) {
 61     if (location == Location.INTERIOR)
 62       this.depth[geomIndex][posIndex]++;
 63   };
 64 
 65   /**
 66    * A Depth object is null (has never been initialized) if all depths are null.
 67    */
 68   jsts.geomgraph.Depth.prototype.isNull = function() {
 69     if (arguments.length > 1) {
 70       this.isNull2.apply(this, arguments);
 71     }
 72 
 73     for (var i = 0; i < 2; i++) {
 74       for (var j = 0; j < 3; j++) {
 75         if (this.depth[i][j] != jsts.geomgraph.Depth.NULL_VALUE)
 76           return false;
 77       }
 78     }
 79     return true;
 80   };
 81 
 82   jsts.geomgraph.Depth.prototype.isNull2 = function(geomIndex) {
 83     if (arguments.length > 2) {
 84       this.isNull3.apply(this, arguments);
 85     }
 86 
 87     return this.depth[geomIndex][1] == jsts.geomgraph.Depth.NULL_VALUE;
 88   };
 89 
 90   jsts.geomgraph.Depth.prototype.isNull3 = function(geomIndex, posIndex) {
 91     return this.depth[geomIndex][posIndex] == jsts.geomgraph.Depth.NULL_VALUE;
 92   };
 93 
 94   jsts.geomgraph.Depth.prototype.add = function(lbl) {
 95     for (var i = 0; i < 2; i++) {
 96       for (var j = 1; j < 3; j++) {
 97         var loc = lbl.getLocation(i, j);
 98         if (loc == Location.EXTERIOR || loc == Location.INTERIOR) {
 99           // initialize depth if it is null, otherwise add this location value
100           if (this.isNull(i, j)) {
101             this.depth[i][j] = jsts.geomgraph.Depth.depthAtLocation(loc);
102           } else
103             this.depth[i][j] += jsts.geomgraph.Depth.depthAtLocation(loc);
104         }
105       }
106     }
107   };
108 
109   jsts.geomgraph.Depth.prototype.getDelta = function(geomIndex) {
110     return this.depth[geomIndex][Position.RIGHT] -
111         this.depth[geomIndex][Position.LEFT];
112   };
113 
114   /**
115    * Normalize the depths for each geometry, if they are non-null. A normalized
116    * depth has depth values in the set { 0, 1 }. Normalizing the depths involves
117    * reducing the depths by the same amount so that at least one of them is 0.
118    * If the remaining value is > 0, it is set to 1.
119    */
120   jsts.geomgraph.Depth.prototype.normalize = function() {
121     for (var i = 0; i < 2; i++) {
122       if (!this.isNull(i)) {
123         var minDepth = this.depth[i][1];
124         if (this.depth[i][2] < minDepth)
125           minDepth = this.depth[i][2];
126 
127         if (minDepth < 0)
128           minDepth = 0;
129         for (var j = 1; j < 3; j++) {
130           var newValue = 0;
131           if (this.depth[i][j] > minDepth)
132             newValue = 1;
133           this.depth[i][j] = newValue;
134         }
135       }
136     }
137   };
138 
139   jsts.geomgraph.Depth.prototype.toString = function() {
140     return;
141     'A: ' + this.depth[0][1] + ',' + this.depth[0][2] + ' B: ' +
142         this.depth[1][1] + ',' + this.depth[1][2];
143   };
144 
145 })();
