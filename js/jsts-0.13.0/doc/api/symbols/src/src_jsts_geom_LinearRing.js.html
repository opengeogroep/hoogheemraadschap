  7 (function() {
  8 
  9   /**
 10    * Models an OGC SFS <code>LinearRing</code>. A LinearRing is a LineString
 11    * which is both closed and simple. In other words, the first and last
 12    * coordinate in the ring must be equal, and the interior of the ring must not
 13    * self-intersect. Either orientation of the ring is allowed.
 14    * <p>
 15    * A ring must have either 0 or 4 or more points. The first and last points
 16    * must be equal (in 2D). If these conditions are not met, the constructors
 17    * throw an {@link IllegalArgumentException}
 18    *
 19    * @requires jsts/geom/LineString.js
 20    */
 21 
 22 
 23   /**
 24    * @extends jsts.geom.LineString
 25    * @constructor
 26    */
 27   jsts.geom.LinearRing = function() {
 28     jsts.geom.LineString.apply(this, arguments);
 29   };
 30   jsts.geom.LinearRing.prototype = OpenLayers.Geometry.LinearRing.prototype;
 31 
 32   for (key in jsts.geom.LineString.prototype) {
 33     jsts.geom.LinearRing.prototype[key] = jsts.geom.LinearRing.prototype[key] ? jsts.geom.LinearRing.prototype[key]
 34         : jsts.geom.LineString.prototype[key];
 35   }
 36 
 37   /**
 38    * Returns <code>Dimension.FALSE</code>, since by definition LinearRings do
 39    * not have a boundary.
 40    *
 41    * @return {int} Dimension.FALSE.
 42    */
 43   jsts.geom.LinearRing.prototype.getBoundaryDimension = function() {
 44     return jsts.geom.Dimension.FALSE;
 45   };
 46 
 47 
 48   /**
 49    * Returns <code>true</code>, since by definition LinearRings are always
 50    * simple.
 51    *
 52    * @return {Boolean} <code>true.</code>
 53    *
 54    * @see Geometry#isSimple
 55    */
 56   jsts.geom.LinearRing.prototype.isSimple = function() {
 57     return true;
 58   };
 59 
 60 
 61   /**
 62    * @return {String} String representation of LinearRing type.
 63    */
 64   jsts.geom.LinearRing.prototype.getGeometryType = function() {
 65     return 'LinearRing';
 66   };
 67 
 68   jsts.geom.LinearRing.MINIMUM_VALID_SIZE = 4;
 69 
 70   OpenLayers.Geometry.LinearRing = jsts.geom.LinearRing;
 71 
 72 })();
