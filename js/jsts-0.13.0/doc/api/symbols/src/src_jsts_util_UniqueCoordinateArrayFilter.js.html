  7 (function() {
  8 
  9   /**
 10    * @requires jsts/geom/CoordinateFilter.js
 11    */
 12 
 13   var ArrayList = javascript.util.ArrayList;
 14   var TreeSet = javascript.util.TreeSet;
 15 
 16   var CoordinateFilter = jsts.geom.CoordinateFilter;
 17 
 18   /**
 19    * A {@link CoordinateFilter} that builds a set of <code>Coordinate</code>s.
 20    * The set of coordinates contains no duplicate points.
 21    *
 22    * @constructor
 23    */
 24   jsts.util.UniqueCoordinateArrayFilter = function() {
 25     this.treeSet = new TreeSet();
 26     this.list = new ArrayList();
 27   };
 28 
 29 
 30   jsts.util.UniqueCoordinateArrayFilter.prototype = new CoordinateFilter();
 31 
 32   jsts.util.UniqueCoordinateArrayFilter.prototype.treeSet = null;
 33   jsts.util.UniqueCoordinateArrayFilter.prototype.list = null;
 34 
 35 
 36   /**
 37    * Returns the gathered <code>Coordinate</code>s.
 38    *
 39    * @return the <code>Coordinate</code>s collected by this
 40    *         <code>CoordinateArrayFilter.</code>
 41    */
 42   jsts.util.UniqueCoordinateArrayFilter.prototype.getCoordinates = function() {
 43     return this.list.toArray();
 44   };
 45 
 46   jsts.util.UniqueCoordinateArrayFilter.prototype.filter = function(coord) {
 47     if (!this.treeSet.contains(coord)) {
 48       this.list.add(coord);
 49       this.treeSet.add(coord);
 50     }
 51   };
 52 })();
