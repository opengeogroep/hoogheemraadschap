  7 /**
  8  * The base class for all geometric objects.
  9  *
 10  *  <H3>Binary Predicates</H3>
 11  * Because it is not clear at this time
 12  * what semantics for spatial
 13  *  analysis methods involving <code>GeometryCollection</code>s would be useful,
 14  *  <code>GeometryCollection</code>s are not supported as arguments to binary
 15  *  predicates (other than <code>convexHull</code>) or the <code>relate</code>
 16  *  method.
 17  *
 18  *  <H3>Set-Theoretic Methods</H3>
 19  *
 20  *  The spatial analysis methods will
 21  *  return the most specific class possible to represent the result. If the
 22  *  result is homogeneous, a <code>Point</code>, <code>LineString</code>, or
 23  *  <code>Polygon</code> will be returned if the result contains a single
 24  *  element; otherwise, a <code>MultiPoint</code>, <code>MultiLineString</code>,
 25  *  or <code>MultiPolygon</code> will be returned. If the result is
 26  *  heterogeneous a <code>GeometryCollection</code> will be returned. <P>
 27  *
 28  *  Because it is not clear at this time what semantics for set-theoretic
 29  *  methods involving <code>GeometryCollection</code>s would be useful,
 30  * <code>GeometryCollections</code>
 31  *  are not supported as arguments to the set-theoretic methods.
 32  *
 33  *  <H4>Representation of Computed Geometries </H4>
 34  *
 35  *  The SFS states that the result
 36  *  of a set-theoretic method is the "point-set" result of the usual
 37  *  set-theoretic definition of the operation (SFS 3.2.21.1). However, there are
 38  *  sometimes many ways of representing a point set as a <code>Geometry</code>.
 39  *  <P>
 40  *
 41  *  The SFS does not specify an unambiguous representation of a given point set
 42  *  returned from a spatial analysis method. One goal of JTS is to make this
 43  *  specification precise and unambiguous. JTS will use a canonical form for
 44  *  <code>Geometry</code>s returned from spatial analysis methods. The canonical
 45  *  form is a <code>Geometry</code> which is simple and noded:
 46  *  <UL>
 47  *    <LI> Simple means that the Geometry returned will be simple according to
 48  *    the JTS definition of <code>isSimple</code>.
 49  *    <LI> Noded applies only to overlays involving <code>LineString</code>s. It
 50  *    means that all intersection points on <code>LineString</code>s will be
 51  *    present as endpoints of <code>LineString</code>s in the result.
 52  *  </UL>
 53  *  This definition implies that non-simple geometries which are arguments to
 54  *  spatial analysis methods must be subjected to a line-dissolve process to
 55  *  ensure that the results are simple.
 56  *
 57  *  <H4> Constructed Points And The Precision Model </H4>
 58  *
 59  *  The results computed by the set-theoretic methods may contain constructed
 60  *  points which are not present in the input <code>Geometry</code>.
 61  *  These new points arise from intersections between line segments in the
 62  *  edges of the input <code>Geometry</code>s. In the general case it is not
 63  *  possible to represent constructed points exactly. This is due to the fact
 64  *  that the coordinates of an intersection point may contain twice as many bits
 65  *  of precision as the coordinates of the input line segments. In order to
 66  *  represent these constructed points explicitly, JTS must truncate them to fit
 67  *  the <code>PrecisionModel</code>.
 68  *
 69  *  Unfortunately, truncating coordinates moves them slightly. Line segments
 70  *  which would not be coincident in the exact result may become coincident in
 71  *  the truncated representation. This in turn leads to "topology collapses" --
 72  *  situations where a computed element has a lower dimension than it would in
 73  *  the exact result.
 74  *
 75  *  When JTS detects topology collapses during the computation of spatial
 76  *  analysis methods, it will throw an exception. If possible the exception will
 77  *  report the location of the collapse.
 78  *
 79  *  #equals(Object) and #hashCode are not overridden, so that when two
 80  *  topologically equal Geometries are added to HashMaps and HashSets, they
 81  *  remain distinct. This behaviour is desired in many cases.
 82  */

257  * Tests whether this {@link Geometry} is simple. In general, the SFS

259  * <UL>

279   this.checkNotGeometryCollection(this);
280   var op = new jsts.operation.IsSimpleOp(this);
281   return op.isSimple();

372 /**
373  * Computes the centroid of this <code>Geometry</code>. The centroid is equal
374  * to the centroid of the set of component Geometries of highest dimension
375  * (since the lower-dimension geometries contribute zero "weight" to the
376  * centroid)
377  *
378  * @return a {@link Point} which is the centroid of this Geometry.
379  */
380 jsts.geom.Geometry.prototype.getCentroid = function() {
381   if (this.isEmpty()) {
382     return null;
383   }
384   var cent;
385   var centPt = null;
386   var dim = this.getDimension();
387   if (dim === 0) {
388     cent = new jsts.algorithm.CentroidPoint();
389     cent.add(this);
390     centPt = cent.getCentroid();
391   } else if (dim === 1) {
392     cent = new jsts.algorithm.CentroidLine();
393     cent.add(this);
394     centPt = cent.getCentroid();
395   } else {
396     cent = new jsts.algorithm.CentroidArea();
397     cent.add(this);
398     centPt = cent.getCentroid();
399   }
400   return this.createPointFromInternalCoord(centPt, this);
401 
402 };

536 /**
537  * Tests whether this geometry touches the specified geometry.
538  * <p>
539  * The <code>touches</code> predicate has the following equivalent
540  * definitions:
541  * <ul>
542  * <li>The geometries have at least one point in common, but their interiors do
543  * not intersect.
544  * <li>The DE-9IM Intersection Matrix for the two geometries matches
545  * <code>[FT*******]</code> or <code>[F**T*****]</code> or
546  * <code>[F***T****]</code>
547  * </ul>
548  * If both geometries have dimension 0, this predicate returns
549  * <code>false</code>
550  *
551  * @param {Geometry}
552  *          g the <code>Geometry</code> with which to compare this
553  *          <code>Geometry.</code>
554  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
555  *         touch; Returns <code>false</code> if both <code>Geometry</code>s
556  *         are points.
557  */
558 jsts.geom.Geometry.prototype.touches = function(g) {
559   // short-circuit test
560   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {
561     return false;
562   }
563   return this.relate(g).isTouches(this.getDimension(), g.getDimension());
564 };

567 /**
568  * Tests whether this geometry intersects the specified geometry.
569  * <p>
570  * The <code>intersects</code> predicate has the following equivalent
571  * definitions:
572  * <ul>
573  * <li>The two geometries have at least one point in common
574  * <li>The DE-9IM Intersection Matrix for the two geometries matches
575  * <code>[T********]</code> or <code>[*T*******]</code> or
576  * <code>[***T*****]</code> or <code>[****T****]</code>
577  * <li><code>! g.disjoint(this)</code> (<code>intersects</code> is the
578  * inverse of <code>disjoint</code>)
579  * </ul>
580  *
581  * @param {Geometry}
582  *          g the <code>Geometry</code> with which to compare this
583  *          <code>Geometry.</code>
584  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
585  *         intersect.
586  *
587  * @see Geometry#disjoint
588  */
589 jsts.geom.Geometry.prototype.intersects = function(g) {
590 
591   // short-circuit envelope test
592   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {
593     return false;
594   }
595 
596   /**
597    * TODO: (MD) Add optimizations: - for P-A case: If P is in env(A), test for
598    * point-in-poly - for A-A case: If env(A1).overlaps(env(A2)) test for
599    * overlaps via point-in-poly first (both ways) Possibly optimize selection of
600    * point to test by finding point of A1 closest to centre of env(A2). (Is
601    * there a test where we shouldn't bother - e.g. if env A is much smaller than
602    * env B, maybe there's no point in testing pt(B) in env(A)?
603    */
604 
605   // optimization for rectangle arguments
606   if (this.isRectangle()) {
607     return RectangleIntersects.intersects(this, g);
608   }
609   if (g.isRectangle()) {
610     return RectangleIntersects.intersects(g, this);
611   }
612   // general case
613   return this.relate(g).isIntersects();
614 };

617 /**
618  * Tests whether this geometry crosses the specified geometry.
619  * <p>
620  * The <code>crosses</code> predicate has the following equivalent
621  * definitions:
622  * <ul>
623  * <li>The geometries have some but not all interior points in common.
624  * <li>The DE-9IM Intersection Matrix for the two geometries matches
625  * <ul>
626  * <li><code>[T*T******]</code> (for P/L, P/A, and L/A situations)
627  * <li><code>[T*****T**]</code> (for L/P, A/P, and A/L situations)
628  * <li><code>[0********]</code> (for L/L situations)
629  * </ul>
630  * </ul>
631  * For any other combination of dimensions this predicate returns
632  * <code>false</code>.
633  * <p>
634  * The SFS defined this predicate only for P/L, P/A, L/L, and L/A situations.
635  * JTS extends the definition to apply to L/P, A/P and A/L situations as well,
636  * in order to make the relation symmetric.
637  *
638  * @param {Geometry}
639  *          g the <code>Geometry</code> with which to compare this
640  *          <code>Geometry.</code>
641  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
642  *         cross.
643  */
644 jsts.geom.Geometry.prototype.crosses = function(g) {
645   // short-circuit test

683  * Tests whether this geometry contains the specified geometry.
684  * <p>
685  * The <code>contains</code> predicate has the following equivalent
686  * definitions:
687  * <ul>
688  * <li>Every point of the other geometry is a point of this geometry, and the
689  * interiors of the two geometries have at least one point in common.
690  * <li>The DE-9IM Intersection Matrix for the two geometries matches
691  * <code>[T*****FF*]</code>
692  * <li><code>g.within(this)</code> (<code>contains</code> is the converse
693  * of <code>within</code>)
694  * </ul>
695  * An implication of the definition is that "Geometries do not contain their
696  * boundary". In other words, if a geometry A is a subset of the points in the
697  * boundary of a geometry B, <code>B.contains(A) = false</code>

699  * @param {Geometry}
700  *          g the <code>Geometry</code> with which to compare this
701  *          <code>Geometry.</code>

754  * Tests whether this geometry covers the specified geometry.
755  * <p>
756  * The <code>covers</code> predicate has the following equivalent definitions:
757  * <ul>
758  * <li>Every point of the other geometry is a point of this geometry.
759  * <li>The DE-9IM Intersection Matrix for the two geometries matches
760  * <code>[T*****FF*]</code> or <code>[*T****FF*]</code> or
761  * <code>[***T**FF*]</code> or <code>[****T*FF*]</code>
762  * <li><code>g.coveredBy(this)</code> (<code>covers</code> is the converse
763  * of <code>coveredBy</code>)
764  * </ul>
765  * If either geometry is empty, the value of this predicate is <tt>false</tt>.
766  * <p>
767  * This predicate is similar to {@link #contains}, but is more inclusive (i.e.
768  * returns <tt>true</tt> for more cases). In particular, unlike
769  * <code>contains</code> it does not distinguish between points in the
770  * boundary and in the interior of geometries. For most situations,
771  * <code>covers</code> should be used in preference to <code>contains</code>.
772  * As an added benefit, <code>covers</code> is more amenable to optimization,
773  * and hence should be more performant.

776  *          g the <code>Geometry</code> with which to compare this
777  *          <code>Geometry.</code>
778  * @return {boolean} <code>true</code> if this <code>Geometry</code> covers
779  *         <code>g.</code>

781  * @see Geometry#contains
782  * @see Geometry#coveredBy
783  */
784 jsts.geom.Geometry.prototype.covers = function(g) {
785   // short-circuit test
786   if (!this.getEnvelopeInternal().covers(g.getEnvelopeInternal())) {
787     return false;
788   }
789   // optimization for rectangle arguments
790   if (this.isRectangle()) {
791     // since we have already tested that the test envelope is covered
792     return true;
793   }
794   return this.relate(g).isCovers();
795 };

1280 /**
1281  * Returns whether this <code>Geometry</code> is greater than, equal to, or
1282  * less than another <code>Geometry</code>.
1283  * <P>
1284  *
1285  * If their classes are different, they are compared using the following
1286  * ordering:
1287  * <UL>
1288  * <LI> Point (lowest)
1289  * <LI> MultiPoint
1290  * <LI> LineString
1291  * <LI> LinearRing
1292  * <LI> MultiLineString
1293  * <LI> Polygon
1294  * <LI> MultiPolygon
1295  * <LI> GeometryCollection (highest)
1296  * </UL>
1297  * If the two <code>Geometry</code>s have the same class, their first
1298  * elements are compared. If those are the same, the second elements are
1299  * compared, etc.
1300  *
1301  * @param {Geometry}
1302  *          other a <code>Geometry</code> with which to compare this
1303  *          <code>Geometry.</code>
1304  * @return {number} a positive number, 0, or a negative number, depending on
1305  *         whether this object is greater than, equal to, or less than
1306  *         <code>o</code>, as defined in "Normal Form For Geometry" in the
1307  *         JTS Technical Specifications.
1308  */
1309 jsts.geom.Geometry.prototype.compareTo = function(o) {
1310   var other = o;
1311   if (this.getClassSortIndex() !== other.getClassSortIndex()) {
1312     return this.getClassSortIndex() - other.getClassSortIndex();
1313   }
1314   if (this.isEmpty() && other.isEmpty()) {
1315     return 0;
1316   }
1317   if (this.isEmpty()) {
1318     return -1;
1319   }
1320   if (other.isEmpty()) {
1321     return 1;
1322   }
1323   return this.compareToSameClass(o);
1324 };

1326 /**
1327  * Returns whether the two <code>Geometry</code>s are equal, from the point
1328  * of view of the <code>equalsExact</code> method. Called by
1329  * <code>equalsExact</code> . In general, two <code>Geometry</code> classes
1330  * are considered to be "equivalent" only if they are the same class. An
1331  * exception is <code>LineString</code> , which is considered to be equivalent
1332  * to its subclasses.
1333  *
1334  * @param {Geometry}
1335  *          other the <code>Geometry</code> with which to compare this
1336  *          <code>Geometry</code> for equality.
1337  * @return {boolean} <code>true</code> if the classes of the two
1338  *         <code>Geometry</code> s are considered to be equal by the
1339  *         <code>equalsExact</code> method.
1340  */
1341 jsts.geom.Geometry.prototype.isEquivalentClass = function(other) {
1342   if (this instanceof jsts.geom.Point && other instanceof jsts.geom.Point) {
1343     return true;
1344   } else if (this instanceof jsts.geom.LineString &&
1345       (other instanceof jsts.geom.LineString | other instanceof jsts.geom.LinearRing)) {
1346     return true;
1347   } else if (this instanceof jsts.geom.LinearRing &&
1348       (other instanceof jsts.geom.LineString | other instanceof jsts.geom.LinearRing)) {
1349     return true;
1350   } else if (this instanceof jsts.geom.Polygon &&
1351       (other instanceof jsts.geom.Polygon)) {
1352     return true;
1353   } else if (this instanceof jsts.geom.MultiPoint &&
1354       (other instanceof jsts.geom.MultiPoint)) {
1355     return true;
1356   } else if (this instanceof jsts.geom.MultiLineString &&
1357       (other instanceof jsts.geom.MultiLineString)) {
1358     return true;
1359   } else if (this instanceof jsts.geom.MultiPolygon &&
1360       (other instanceof jsts.geom.MultiPolygon)) {
1361     return true;
1362   } else if (this instanceof jsts.geom.GeometryCollection &&
1363       (other instanceof jsts.geom.GeometryCollection)) {
1364     return true;
1365   }
1366 
1367   return false;
1368 };

1431 /**
1432  * Returns the first non-zero result of <code>compareTo</code> encountered as
1433  * the two <code>Collection</code>s are iterated over. If, by the time one of
1434  * the iterations is complete, no non-zero result has been encountered, returns
1435  * 0 if the other iteration is also complete. If <code>b</code> completes
1436  * before <code>a</code>, a positive number is returned; if a before b, a
1437  * negative number.
1438  *
1439  * @param {Array}
1440  *          a a <code>Collection</code> of <code>Comparable</code>s.
1441  * @param {Array}
1442  *          b a <code>Collection</code> of <code>Comparable</code>s.
1443  * @return {number} the first non-zero <code>compareTo</code> result, if any;
1444  *         otherwise, zero.
1445  */
1446 jsts.geom.Geometry.prototype.compare = function(a, b) {
1447   var i = a.iterator();
1448   var j = b.iterator();
1449   while (i.hasNext() && j.hasNext()) {
1450     var aElement = i.next();
1451     var bElement = j.next();
1452     var comparison = aElement.compareTo(bElement);
1453     if (comparison !== 0) {
1454       return comparison;
1455     }
1456   }
1457   if (i.hasNext()) {
1458     return 1;
1459   }
1460   if (j.hasNext()) {
1461     return -1;
1462   }
1463   return 0;
1464 };

1470  * @param {jsts.geom.Coordinate}

1472  * @param {number}
1473  *          tolerance tolerance when comparing.
1474  * @return {boolean} true if equal.
