  7 /**
  8  * Base class for STRtree and SIRtree. STR-packed R-trees are described in:
  9  * P. Rigaux, Michel Scholl and Agnes Voisard. Spatial Databases With
 10  * Application To GIS. Morgan Kaufmann, San Francisco, 2002.
 11  * <p>
 12  * This implementation is based on Boundables rather than just AbstractNodes,
 13  * because the STR algorithm operates on both nodes and
 14  * data, both of which are treated here as Boundables.

 16  * @see STRtree
 17  * @see SIRtree

 97 /**
 98  * Creates parent nodes, grandparent nodes, and so forth up to the root node,
 99  * for the data that has been inserted into the tree. Can only be called once,
100  * and thus can be called only after all of the data has been inserted into the
101  * tree.
102  */
103 jsts.index.strtree.AbstractSTRtree.prototype.build = function() {
104   jsts.util.Assert.isTrue(!this.built);
105   this.root = this.itemBoundables.length === 0 ? this.createNode(0) : this
106       .createHigherLevels(this.itemBoundables, -1);
107   this.built = true;
108 };

121 /**
122  * Sorts the childBoundables then divides them into groups of size M, where M is
123  * the node capacity.
124  */
125 jsts.index.strtree.AbstractSTRtree.prototype.createParentBoundables = function(
126     childBoundables, newLevel) {
127   jsts.util.Assert.isTrue(!(childBoundables.length === 0));
128   var parentBoundables = [];
129   parentBoundables.push(this.createNode(newLevel));
130   var sortedChildBoundables = [];
131   for (var i = 0; i < childBoundables.length; i++) {
132     sortedChildBoundables.push(childBoundables[i]);
133   }
134   sortedChildBoundables.sort(this.getComparator());
135   for (var i = 0; i < sortedChildBoundables.length; i++) {
136     var childBoundable = sortedChildBoundables[i];
137     if (this.lastNode(parentBoundables).getChildBoundables().length === this
138         .getNodeCapacity()) {
139       parentBoundables.push(this.createNode(newLevel));
140     }
141     this.lastNode(parentBoundables).addChildBoundable(childBoundable);
142   }
143   return parentBoundables;
144 };

169 /**
170  * Creates the levels higher than the given level
171  *
172  * @param {Array}
173  *          boundablesOfALevel the level to build on.
174  * @param {number}
175  *          level the level of the Boundables, or -1 if the boundables are item
176  *          boundables (that is, below level 0).
177  * @return {jsts.index.strtree.AbstractNode} the root, which may be a ParentNode
178  *         or a LeafNode.
179  * @private
180  */
181 jsts.index.strtree.AbstractSTRtree.prototype.createHigherLevels = function(
182     boundablesOfALevel, level) {
183   jsts.util.Assert.isTrue(!(boundablesOfALevel.length === 0));
184   var parentBoundables = this.createParentBoundables(boundablesOfALevel,
185       level + 1);
186   if (parentBoundables.length === 1) {
187     return parentBoundables[0];
188   }
189   return this.createHigherLevels(parentBoundables, level + 1);
190 };

214  * @return {number}
215  */
216 jsts.index.strtree.AbstractSTRtree.prototype.size = function() {
217   if (arguments.length === 1) {
218     return this.size2(arguments[0]);
219   }
220 
221   if (!this.built) {
222     this.build();
223   }
224   if (this.itemBoundables.length === 0) {
225     return 0;
226   }
227   return this.size2(root);
228 };

230 /**
231  * @param {jsts.index.strtree.AbstractNode=}
232  *          [node].
233  * @return {number}
234  */
235 jsts.index.strtree.AbstractSTRtree.prototype.size2 = function(node) {
236   var size = 0;
237   var childBoundables = node.getChildBoundables();
238   for (var i = 0; i < childBoundables.length; i++) {
239     var childBoundable = childBoundables[i];
240     if (childBoundable instanceof jsts.index.strtree.AbstractNode) {
241       size += this.size(childBoundable);
242     } else if (childBoundable instanceof jsts.index.strtree.ItemBoundable) {
243       size += 1;
244     }
245   }
246   return size;
247 };

267 /**
268  * @param {jsts.index.strtree.AbstractNode}
269  *          [node].
270  * @return {number}
271  */
272 jsts.index.strtree.AbstractSTRtree.prototype.depth2 = function() {
273   var maxChildDepth = 0;
274   var childBoundables = node.getChildBoundables();
275   for (var i = 0; i < childBoundables.length; i++) {
276     var childBoundable = childBoundables[i];
277     if (childBoundable instanceof jsts.index.strtree.AbstractNode) {
278       var childDepth = this.depth(childBoundable);
279       if (childDepth > maxChildDepth)
280         maxChildDepth = childDepth;
281     }
282   }
283   return maxChildDepth + 1;
284 };

299 /**
300  * Also builds the tree, if necessary.
301  *
302  * @param {Object}
303  *          searchBounds
304  * @param {jsts.index.ItemVisitor}
305  *          [visitor].
306  * @param {jsts.index.strtree.AbstractNode}
307  *          [node].
308  * @param {Array}
309  *          [matches].
310  * @return {Array}
311  */
312 jsts.index.strtree.AbstractSTRtree.prototype.query = function(searchBounds) {
313   if (arguments.length > 1) {
314     this.query2.apply(this, arguments);
315   }
316 
317   if (!this.built) {
318     this.build();
319   }
320   var matches = [];
321   if (this.itemBoundables.length === 0) {
322     jsts.util.Assert.isTrue(this.root.getBounds() === null);
323     return matches;
324   }
325   if (this.getIntersectsOp().intersects(this.root.getBounds(), searchBounds)) {
326     this.query3(searchBounds, this.root, matches);
327   }
328   return matches;
329 };

331 jsts.index.strtree.AbstractSTRtree.prototype.query2 = function(searchBounds,
332     visitor) {
333   if (arguments.length > 2) {
334     this.query3.apply(this, arguments);
335   }
336 
337   if (!this.built) {
338     this.build();
339   }
340   if (this.itemBoundables.length === 0) {
341     jsts.util.Assert.isTrue(this.root.getBounds() === null);
342   }
343   if (this.getIntersectsOp().intersects(this.root.getBounds(), searchBounds)) {
344     this.query4(searchBounds, root, visitor);
345   }
346 };

348 /**
349  * @private
350  */
351 jsts.index.strtree.AbstractSTRtree.prototype.query3 = function(searchBounds,
352     node, matches) {
353   if (!(arguments[2] instanceof Array)) {
354     this.query4.apply(this, arguments);
355   }
356 
357   var childBoundables = node.getChildBoundables();
358   for (var i = 0; i < childBoundables.length; i++) {
359     var childBoundable = childBoundables[i];
360     if (!this.getIntersectsOp().intersects(childBoundable.getBounds(),
361         searchBounds)) {
362       continue;
363     }
364     if (childBoundable instanceof jsts.index.strtree.AbstractNode) {
365       this.query3(searchBounds, childBoundable, matches);
366     } else if (childBoundable instanceof jsts.index.strtree.ItemBoundable) {
367       matches.push(childBoundable.getItem());
368     } else {
369       jsts.util.Assert.shouldNeverReachHere();
370     }
371   }
372 };
373 
374 /**
375  * @private
376  */
377 jsts.index.strtree.AbstractSTRtree.prototype.query4 = function(searchBounds,
378     node, visitor) {
379   var childBoundables = node.getChildBoundables();
380   for (var i = 0; i < childBoundables.length; i++) {
381     var childBoundable = childBoundables[i];
382     if (!this.getIntersectsOp().intersects(childBoundable.getBounds(),
383         searchBounds)) {
384       continue;
385     }
386     if (childBoundable instanceof jsts.index.strtree.AbstractNode) {
387       this.query4(searchBounds, childBoundable, visitor);
388     } else if (childBoundable instanceof jsts.index.strtree.ItemBoundable) {
389       visitor.visitItem(childBoundable.getItem());
390     } else {
391       jsts.util.Assert.shouldNeverReachHere();
392     }
393   }
394 };

405 /**
406  * Gets a tree structure (as a nested list) corresponding to the structure of
407  * the items and nodes in this tree.
408  * <p>
409  * The returned {@link List}s contain either {@link Object} items, or Lists
410  * which correspond to subtrees of the tree Subtrees which do not contain any
411  * items are not included.
412  * <p>
413  * Builds the tree if necessary.
414  *
415  * @return {Array} a List of items and/or Lists.
416  */
417 jsts.index.strtree.AbstractSTRtree.prototype.itemsTree = function() {
418   if (arguments.length === 1) {
419     return this.itemsTree2.apply(this, arguments);
420   }
421 
422   if (!this.built) {
423     this.build();
424   }
425 
426   var valuesTree = this.itemsTree2(this.root);
427   if (valuesTree === null)
428     return [];
429   return valuesTree;
430 };
431 
432 jsts.index.strtree.AbstractSTRtree.prototype.itemsTree2 = function(node) {
433   var valuesTreeForNode = [];
434   var childBoundables = node.getChildBoundables();
435   for (var i = 0; i < childBoundables.length; i++) {
436     var childBoundable = childBoundables[i];
437     if (childBoundable instanceof jsts.index.strtree.AbstractNode) {
438       var valuesTreeForChild = this.itemsTree(childBoundable);
439       // only add if not null (which indicates an item somewhere in this tree
440       if (valuesTreeForChild != null)
441         valuesTreeForNode.push(valuesTreeForChild);
442     } else if (childBoundable instanceof jsts.index.strtree.ItemBoundable) {
443       valuesTreeForNode.push(childBoundable.getItem());
444     } else {
445       jsts.util.Assert.shouldNeverReachHere();
446     }
447   }
448   if (valuesTreeForNode.length <= 0)
449     return null;
450   return valuesTreeForNode;
451 };

453 /**
454  * Removes an item from the tree. (Builds the tree, if necessary.)
455  *
456  * @param {Object}
457  *          searchBounds
458  * @param {jsts.index.strtree.AbstractNode}
459  *          [node].
460  * @param {Object]
461  *          item}
462  * @return {boolean}
463  */
464 jsts.index.strtree.AbstractSTRtree.prototype.remove = function(searchBounds,
465     item) {
466   // TODO: argument switch
467 
468 
469   if (!this.built) {
470     this.build();
471   }
472   if (this.itemBoundables.length === 0) {
473     jsts.util.Assert.isTrue(this.root.getBounds() == null);
474   }
475   if (this.getIntersectsOp().intersects(this.root.getBounds(), searchBounds)) {
476     return this.remove2(searchBounds, this.root, item);
477   }
478   return false;
479 };

481 jsts.index.strtree.AbstractSTRtree.prototype.remove2 = function(searchBounds,
482     node, item) {
483   // first try removing item from this node
484   var found = this.removeItem(node, item);
485   if (found)
486     return true;
487 
488   var childToPrune = null;
489   // next try removing item from lower nodes
490   var childBoundables = node.getChildBoundables();
491   for (var i = 0; i < childBoundables.length; i++) {
492     var childBoundable = childBoundables[i];
493     if (!this.getIntersectsOp().intersects(childBoundable.getBounds(),
494         searchBounds)) {
495       continue;
496     }
497     if (childBoundable instanceof jsts.index.strtree.AbstractNode) {
498       found = this.remove(searchBounds, childBoundable, item);
499       // if found, record child for pruning and exit
500       if (found) {
501         childToPrune = childBoundable;
502         break;
503       }
504     }
505   }
506   // prune child if possible
507   if (childToPrune != null) {
508     if (childToPrune.getChildBoundables().length === 0) {
509       childBoundables.splice(childBoundables.indexOf(childToPrune), 1);
510     }
511   }
512   return found;
513 };

517 /**
518  *
519  * @param {jsts.index.strtree.AbstractNode}
520  *          node
521  * @param {Object}
522  *          item
523  * @return {boolean}
524  */
525 jsts.index.strtree.AbstractSTRtree.prototype.removeItem = function(node, item) {
526   var childToRemove = null;
527   var childBoundables = node.getChildBoundables();
528   for (var i = 0; i < childBoundables.length; i++) {
529     var childBoundable = childBoundables[i];
530     if (childBoundable instanceof jsts.index.strtree.ItemBoundable) {
531       if (childBoundable.getItem() === item)
532         childToRemove = childBoundable;
533     }
534   }
535   if (childToRemove !== null) {
536     childBoundables.splice(childBoundables.indexOf(childToRemove), 1);
537     return true;
538   }
539   return false;
540 };

543 jsts.index.strtree.AbstractSTRtree.prototype.boundablesAtLevel = function(level) {
544   if (arguments.length > 1) {
545     this.boundablesAtLevel2.apply(this, arguments);
546     return;
547   }
548 
549   var boundables = [];
550   this.boundablesAtLevel2(level, this.root, boundables);
551   return boundables;
552 };
553 
554 /**
555  * @param {number}
556  *          level
557  * @param {jsts.index.strtree.AbstractNode}
558  *          [top].
559  * @param {Array}
560  *          [boundables].
561  * @return {?Array}
562  */
563 jsts.index.strtree.AbstractSTRtree.prototype.boundablesAtLevel2 = function(
564     level, top, boundables) {
565   jsts.util.Assert.isTrue(level > -2);
566   if (top.getLevel() === level) {
567     boundables.add(top);
568     return;
569   }
570   var childBoundables = node.getChildBoundables();
571   for (var i = 0; i < childBoundables.length; i++) {
572     var boundable = childBoundables[i];
573     if (boundable instanceof jsts.index.strtree.AbstractNode) {
574       this.boundablesAtLevel(level, boundable, boundables);
575     } else {
576       jsts.util.Assert.isTrue(boundable instanceof jsts.index.strtree.ItemBoundable);
577       if (level === -1) {
578         boundables.add(boundable);
579       }
580     }
581   }
582   return;
583 };
