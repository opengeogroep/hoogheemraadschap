 22   /**
 23    * A GeometryGraph is a graph that models a given Geometry
 24    *
 25    * @param {int}
 26    *          argIndex
 27    * @param {Geometry}
 28    *          parentGeom
 29    * @param {BoundaryNodeRule}
 30    *          boundaryNodeRule
 31    * @augments jsts.planargraph.PlanarGraph
 32    */
 33   jsts.geomgraph.GeometryGraph = function(argIndex, parentGeom, boundaryNodeRule) {
 34     jsts.geomgraph.PlanarGraph.call(this);
 35 
 36     this.lineEdgeMap = new javascript.util.HashMap();
 37 
 38     this.ptLocator = new jsts.algorithm.PointLocator();
 39 
 40     this.argIndex = argIndex;
 41     this.parentGeom = parentGeom;
 42     this.boundaryNodeRule = boundaryNodeRule ||
 43         jsts.algorithm.BoundaryNodeRule.OGC_SFS_BOUNDARY_RULE;
 44     if (parentGeom !== null) {
 45       this.add(parentGeom);
 46     }
 47   };
 48

 63    * @param {BoundaryNodeRule}
 64    *          boundaryNodeRule
 65    * @param {int}
 66    *          boundaryCount
 67    * @return {int}
 68    */
 69   jsts.geomgraph.GeometryGraph.determineBoundary = function(boundaryNodeRule, boundaryCount) {
 70     return boundaryNodeRule.isInBoundary(boundaryCount) ? Location.BOUNDARY
 71         : Location.INTERIOR;

185   /**
186    * @param {Geometry}
187    *          g
188    */
189   jsts.geomgraph.GeometryGraph.prototype.add = function(g) {
190     if (g.isEmpty()) {
191       return;
192     }
193 
194     // check if this Geometry should obey the Boundary Determination Rule
195     // all collections except MultiPolygons obey the rule
196     if (g instanceof jsts.geom.MultiPolygon)
197       this.useBoundaryDeterminationRule = false;
198 
199     if (g instanceof jsts.geom.Polygon)
200       this.addPolygon(g);
201     // LineString also handles LinearRings
202     else if (g instanceof jsts.geom.LineString)
203       this.addLineString(g);
204     else if (g instanceof jsts.geom.Point)
205       this.addPoint(g);
206     else if (g instanceof jsts.geom.MultiPoint)
207       this.addCollection(g);
208     else if (g instanceof jsts.geom.MultiLineString)
209       this.addCollection(g);
210     else if (g instanceof jsts.geom.MultiPolygon)
211       this.addCollection(g);
212     else if (g instanceof jsts.geom.GeometryCollection)
213       this.addCollection(g);
214     else
215       throw new jsts.error.IllegalArgumentError('Geometry type not supported.');
216   };

232    * correct.
233    */
234   jsts.geomgraph.GeometryGraph.prototype.addEdge = function(e) {
235     this.insertEdge(e);
236     var coord = e.getCoordinates();
237     // insert the endpoint as a node, to mark that it is on the boundary
238     this.insertPoint(this.argIndex, coord[0], Location.BOUNDARY);
239     this.insertPoint(this.argIndex, coord[coord.length - 1], Location.BOUNDARY);
240   };

252   /**
253    * @param {LineString}
254    *          line
255    * @private
256    */
257   jsts.geomgraph.GeometryGraph.prototype.addLineString = function(line) {
258     var coord = jsts.geom.CoordinateArrays.removeRepeatedPoints(line
259         .getCoordinates());
260 
261     if (coord.length < 2) {
262       this.hasTooFewPoints = true;
263       this.invalidPoint = coords[0];
264       return;
265     }
266 
267     // add the edge for the LineString
268     // line edges do not have locations for their left and right sides
269     var e = new jsts.geomgraph.Edge(coord, new jsts.geomgraph.Label(
270         this.argIndex, Location.INTERIOR));
271     this.lineEdgeMap.put(line, e);
272     this.insertEdge(e);
273     /**
274      * Add the boundary points of the LineString, if any. Even if the LineString
275      * is closed, add both points as if they were endpoints. This allows for the
276      * case that the node already exists and is a boundary point.
277      */
278     Assert.isTrue(coord.length >= 2, 'found LineString with single point');
279     this.insertBoundaryPoint(this.argIndex, coord[0]);
280     this.insertBoundaryPoint(this.argIndex, coord[coord.length - 1]);
281   };

284   /**
285    * Adds a polygon ring to the graph. Empty rings are ignored.
286    *
287    * The left and right topological location arguments assume that the ring is
288    * oriented CW. If the ring is in the opposite orientation, the left and right
289    * locations must be interchanged.
290    *
291    * @private
292    */
293   jsts.geomgraph.GeometryGraph.prototype.addPolygonRing = function(lr, cwLeft, cwRight) {
294     // don't bother adding empty holes
295     if (lr.isEmpty())
296       return;
297 
298     var coord = jsts.geom.CoordinateArrays.removeRepeatedPoints(lr
299         .getCoordinates());
300 
301     if (coord.length < 4) {
302       this.hasTooFewPoints = true;
303       this.invalidPoint = coord[0];
304       return;
305     }
306 
307     var left = cwLeft;
308     var right = cwRight;
309     if (jsts.algorithm.CGAlgorithms.isCCW(coord)) {
310       left = cwRight;
311       right = cwLeft;
312     }
313     var e = new jsts.geomgraph.Edge(coord, new jsts.geomgraph.Label(
314         this.argIndex, Location.BOUNDARY, left, right));
315     this.lineEdgeMap.put(lr, e);
316 
317     this.insertEdge(e);
318     // insert the endpoint as a node, to mark that it is on the boundary
319     this.insertPoint(this.argIndex, coord[0], Location.BOUNDARY);
320   };

323   /**
324    * @private
325    */
326   jsts.geomgraph.GeometryGraph.prototype.addPolygon = function(p) {
327     this.addPolygonRing(p.getExteriorRing(), Location.EXTERIOR,
328         Location.INTERIOR);
329 
330     for (var i = 0; i < p.getNumInteriorRing(); i++) {
331       var hole = p.getInteriorRingN(i);
332 
333       // Holes are topologically labelled opposite to the shell, since
334       // the interior of the polygon lies on their opposite side
335       // (on the left, if the hole is oriented CW)
336       this.addPolygonRing(hole, Location.INTERIOR, Location.EXTERIOR);
337     }
338   };

341   jsts.geomgraph.GeometryGraph.prototype.computeEdgeIntersections = function(g, li,
342       includeProper) {
343     var si = new jsts.geomgraph.index.SegmentIntersector(li, includeProper,
344         true);
345     si.setBoundaryNodes(this.getBoundaryNodes(), g.getBoundaryNodes());
346 
347     var esi = this.createEdgeSetIntersector();
348     esi.computeIntersections(this.edges, g.edges, si);
349 
350     return si;
351   };

354   /**
355    * Compute self-nodes, taking advantage of the Geometry type to minimize the
356    * number of intersection tests. (E.g. rings are not tested for
357    * self-intersection, since they are assumed to be valid).
358    *
359    * @param {LineIntersector}
360    *          li the LineIntersector to use.
361    * @param {boolean}
362    *          computeRingSelfNodes if <false>, intersection checks are optimized
363    *          to not test rings for self-intersection.
364    * @return {SegmentIntersector} the SegmentIntersector used, containing
365    *         information about the intersections found.
366    */
367   jsts.geomgraph.GeometryGraph.prototype.computeSelfNodes = function(li, computeRingSelfNodes) {
368     var si = new jsts.geomgraph.index.SegmentIntersector(li, true, false);
369     var esi = this.createEdgeSetIntersector();
370     // optimized test for Polygons and Rings
371     if (!computeRingSelfNodes &&
372         (this.parentGeom instanceof jsts.geom.LinearRing ||
373             this.parentGeom instanceof jsts.geom.Polygon || this.parentGeom instanceof jsts.geom.MultiPolygon)) {
374       esi.computeIntersections(this.edges, si, false);
375     } else {
376       esi.computeIntersections(this.edges, si, true);
377     }
378     this.addSelfIntersectionNodes(this.argIndex);
379     return si;
380   };

383   /**
384    * @private
385    */
386   jsts.geomgraph.GeometryGraph.prototype.insertPoint = function(argIndex, coord, onLocation) {
387     var n = this.nodes.addNode(coord);
388     var lbl = n.getLabel();
389     if (lbl == null) {
390       n.label = new jsts.geomgraph.Label(argIndex, onLocation);
391     } else
392       lbl.setLocation(argIndex, onLocation);
393   };

396   /**
397    * Adds candidate boundary points using the current {@link BoundaryNodeRule}.
398    * This is used to add the boundary points of dim-1 geometries
399    * (Curves/MultiCurves).
400    */
401   jsts.geomgraph.GeometryGraph.prototype.insertBoundaryPoint = function(argIndex, coord) {
402     var n = this.nodes.addNode(coord);
403     var lbl = n.getLabel();
404     // the new point to insert is on a boundary
405     var boundaryCount = 1;
406     // determine the current location for the point (if any)
407     var loc = Location.NONE;
408     if (lbl !== null)
409       loc = lbl.getLocation(argIndex, Position.ON);
410     if (loc === Location.BOUNDARY)
411       boundaryCount++;
412 
413     // determine the boundary status of the point according to the Boundary
414     // Determination Rule
415     var newLoc = jsts.geomgraph.GeometryGraph.determineBoundary(this.boundaryNodeRule,
416         boundaryCount);
417     lbl.setLocation(argIndex, newLoc);
418   };

421   /**
422    * add edge intersections as self intersections from each edge intersection
423    * list
424    *
425    * @param argIndex
426    * @private
427    */
428   jsts.geomgraph.GeometryGraph.prototype.addSelfIntersectionNodes = function(argIndex) {
429     for (var i = this.edges.iterator(); i.hasNext();) {
430       var e = i.next();
431       var eLoc = e.getLabel().getLocation(argIndex);
432       for (var eiIt = e.eiList.iterator(); eiIt.hasNext();) {
433         var ei = eiIt.next();
434         this.addSelfIntersectionNode(argIndex, ei.coord, eLoc);
435       }
436     }
437   };

440   /**
441    * Add a node for a self-intersection. If the node is a potential boundary
442    * node (e.g. came from an edge which is a boundary) then insert it as a
443    * potential boundary node. Otherwise, just add it as a regular node.
444    *
445    * @private
446    */
447   jsts.geomgraph.GeometryGraph.prototype.addSelfIntersectionNode = function(argIndex, coord,
448       loc) {
449     // if this node is already a boundary node, don't change it
450     if (this.isBoundaryNode(argIndex, coord))
451       return;
452     if (loc === Location.BOUNDARY && this.useBoundaryDeterminationRule)
453       this.insertBoundaryPoint(argIndex, coord);
454     else
455       this.insertPoint(argIndex, coord, loc);
456   };
