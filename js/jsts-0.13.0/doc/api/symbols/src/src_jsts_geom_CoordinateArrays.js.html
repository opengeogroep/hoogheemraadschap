 17 /**
 18  * If the coordinate array argument has repeated points, constructs a new array
 19  * containing no repeated points. Otherwise, returns the argument.
 20  *
 21  * @return {Coordinate[]}
 22  * @see #hasRepeatedPoints(Coordinate[])
 23  */
 24 jsts.geom.CoordinateArrays.removeRepeatedPoints = function(coord) {
 25   var coordList;
 26   if (!this.hasRepeatedPoints(coord)) {
 27     return coord;
 28   }
 29   coordList = new jsts.geom.CoordinateList(coord, false);
 30   return coordList;
 31 };

 34 /**
 35  * Returns whether #equals returns true for any two consecutive Coordinates in
 36  * the given array.
 37  *
 38  * @param {Coordinate[]}
 39  *          coord
 40  * @return {boolean}
 41  */
 42 jsts.geom.CoordinateArrays.hasRepeatedPoints = function(coord) {
 43   var i;
 44   for (i = 1; i < coord.length; i++) {
 45     if (coord[i - 1].equals(coord[i])) {
 46       return true;
 47     }
 48   }
 49   return false;
 50 };

 52 /**
 53  * Determines which orientation of the {@link Coordinate} array
 54  * is (overall) increasing.
 55  * In other words, determines which end of the array is "smaller"
 56  * (using the standard ordering on {@link Coordinate}).
 57  * Returns an integer indicating the increasing direction.
 58  * If the sequence is a palindrome, it is defined to be
 59  * oriented in a positive direction.
 60  *
 61  * @param pts the array of Coordinates to test.
 62  * @return <code>1</code> if the array is smaller at the start
 63  * or is a palindrome,
 64  * <code>-1</code> if smaller at the end.
 65  */
 66 jsts.geom.CoordinateArrays.increasingDirection = function(pts) {
 67   for (var i = 0; i < parseInt(pts.length / 2); i++) {
 68     var j = pts.length - 1 - i;
 69     // skip equal points on both ends
 70     var comp = pts[i].compareTo(pts[j]);
 71     if (comp != 0)
 72       return comp;
 73   }
 74   // array must be a palindrome - defined to be in positive direction
 75   return 1;
 76 };
