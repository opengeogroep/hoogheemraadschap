  9 /**
 10  * Constructs a new list from an array of Coordinates, allowing caller to
 11  * specify if repeated points are to be removed.
 12  *
 13  * In JSTS CoordinateList do *not* expose an API extended from ArrayList in Java, instead it is based on a simple JavaScript array.
 14  * CoordinateList does not seem to be used extensively which would warrant keeping the API.
 15  *
 16  * @param {Array.<jsts.geom.Coordinate>}
 17  *          coord the array of coordinates to load into the list.
 18  * @param {boolean}
 19  *          allowRepeated if <code>false</code>, repeated points are removed.
 20  *
 21  * @constructor
 22  */
 23 jsts.geom.CoordinateList = function(coord, allowRepeated) {
 24   this.add(coord, allowRepeated);
 25 };
 26 
 27 jsts.geom.CoordinateList.prototype = new Array();
 28 
 29 
 30 /**
 31  * Adds an array of coordinates to the list.
 32  *
 33  * @param {Array.<jsts.geom.Coordinate>}
 34  *          coord The coordinates.
 35  * @param {boolean}
 36  *          allowRepeated if set to false, repeated coordinates are collapsed.
 37  * @param {boolean}
 38  *          direction if false, the array is added in reverse order.
 39  * @return {boolean} true (as by general collection contract).
 40  */
 41 jsts.geom.CoordinateList.prototype.add = function(coord, allowRepeated,
 42     direction) {
 43   direction = direction || true;
 44 
 45   if (direction) {
 46     for (var i = 0; i < coord.length; i++) {
 47       this.addCoordinate(coord[i], allowRepeated);
 48     }
 49   } else {
 50     for (var i = coord.length - 1; i >= 0; i--) {
 51       this.addCoordinate(coord[i], allowRepeated);
 52     }
 53   }
 54   return true;
 55 };
 56 
 57 
 58 /**
 59  * Adds a coordinate to the end of the list.
 60  *
 61  * @param {Coordinate}
 62  *          coord The coordinates.
 63  * @param {boolean}
 64  *          allowRepeated if set to false, repeated coordinates are collapsed.
 65  */
 66 jsts.geom.CoordinateList.prototype.addCoordinate = function(coord,
 67     allowRepeated) {
 68   // don't add duplicate coordinates
 69   if (!allowRepeated) {
 70     if (this.length >= 1) {
 71       var last = this[this.length - 1];
 72       if (last.equals2D(coord))
 73         return;
 74     }
 75   }
 76   this.push(coord);
 77 };
