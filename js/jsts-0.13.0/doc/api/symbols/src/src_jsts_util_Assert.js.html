  7 (function() {
  8 
  9   /**
 10    * @requires jsts/util/AssertionFailedException.js
 11    */
 12 
 13   var AssertionFailedException = jsts.util.AssertionFailedException;
 14 
 15   /**
 16    * A utility for making programming assertions.
 17    *
 18    * @constructor
 19    * @name jsts.util.Assert
 20    */
 21   jsts.util.Assert = function() {};
 22 
 23   /**
 24    * Throws an <code>AssertionFailedException</code> with the given message if
 25    * the given assertion is not true.
 26    *
 27    * @param {boolean}
 28    *          assertion a condition that is supposed to be true.
 29    * @param {String=}
 30    *          message a description of the assertion.
 31    * @throws AssertionFailedException
 32    *           if the condition is false
 33    */
 34   jsts.util.Assert.isTrue = function(assertion, message) {
 35     if (!assertion) {
 36       if (message === null) {
 37         throw new AssertionFailedException();
 38       } else {
 39         throw new AssertionFailedException(message);
 40       }
 41     }
 42   };
 43 
 44   /**
 45    * Throws an <code>AssertionFailedException</code> with the given message if
 46    * the given objects are not equal, according to the <code>equals</code>
 47    * method.
 48    *
 49    * @param expectedValue
 50    *          the correct value.
 51    * @param actualValue
 52    *          the value being checked.
 53    * @param {string=}
 54    *          message a description of the assertion.
 55    * @throws AssertionFailedException
 56    *           if the two objects are not equal
 57    */
 58   jsts.util.Assert.equals = function(expectedValue, actualValue, message) {
 59     if (!actualValue.equals(expectedValue)) {
 60       throw new AssertionFailedException('Expected ' + expectedValue +
 61           ' but encountered ' + actualValue +
 62           (message != null ? ': ' + message : ''));
 63     }
 64   };
 65 
 66   /**
 67    * Always throws an <code>AssertionFailedException</code> with the given
 68    * message.
 69    *
 70    * @param {string=}
 71    *          message a description of the assertion.
 72    * @throws AssertionFailedException
 73    *           thrown always
 74    */
 75   jsts.util.Assert.shouldNeverReachHere = function(message) {
 76     throw new AssertionFailedException('Should never reach here' +
 77         (message != null ? ': ' + message : ''));
 78   };
 79 
 80 })();
