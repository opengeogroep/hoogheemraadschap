 14   /**
 15    * @extends jsts.geom.Geometry
 16    * @constructor
 17    */
 18   jsts.geom.LineString = function(points) {
 19     OpenLayers.Geometry.Curve.prototype.initialize.apply(this, arguments);
 20 
 21     this.geometries = this.components;
 22   };
 23   jsts.geom.LineString.prototype = OpenLayers.Geometry.LineString.prototype;
 24 
 25   for (key in jsts.geom.Geometry.prototype) {
 26     jsts.geom.LineString.prototype[key] = jsts.geom.Geometry.prototype[key];
 27   }
 28 
 29   /**
 30    * @return {jsts.geom.Coordinate[]} this LineString's internal coordinate
 31    *         array.
 32    */
 33   jsts.geom.LineString.prototype.getCoordinates = function() {
 34     return this.components;
 35   };
 36 
 37 
 38   /**
 39    * @return {jsts.geom.Coordinate} The n'th coordinate of this
 40    *         jsts.geom.LineString.
 41    * @param {int}
 42    *          n index.
 43    */
 44   jsts.geom.LineString.prototype.getCoordinateN = function(n) {
 45     return this.components[n];
 46   };
 47 
 48 
 49   /**
 50    * @return {jsts.geom.Coordinate} The first coordinate of this LineString or
 51    *         null if empty.
 52    */
 53   jsts.geom.LineString.prototype.getCoordinate = function() {
 54     if (this.isEmpty()) {
 55       return null;
 56     }
 57     return this.getCoordinateN(0);
 58   };
 59 
 60 
 61   /**
 62    * @return {int} LineStrings are always 1-dimensional.
 63    */
 64   jsts.geom.LineString.prototype.getDimension = function() {
 65     return 1;
 66   };
 67 
 68 
 69   /**
 70    * @return {int} dimension of the boundary of this jsts.geom.LineString.
 71    */
 72   jsts.geom.LineString.prototype.getBoundaryDimension = function() {
 73     if (this.isClosed()) {
 74       return Dimension.FALSE;
 75     }
 76     return 0;
 77   };
 78 
 79 
 80   /**
 81    * @return {Boolean} true if empty.
 82    */
 83   jsts.geom.LineString.prototype.isEmpty = function() {
 84     return this.components.length === 0;
 85   };
 86 
 87   jsts.geom.LineString.prototype.getNumPoints = function() {
 88     return this.components.length;
 89   };
 90 
 91   jsts.geom.LineString.prototype.getPointN = function(n) {
 92     return this.getFactory().createPoint(this.components[n]);
 93   };
 94 
 95 
 96   jsts.geom.LineString.prototype.getStartPoint = function() {
 97     if (this.isEmpty()) {
 98       return null;
 99     }
100     return this.getPointN(0);
101   };
102 
103   jsts.geom.LineString.prototype.getEndPoint = function() {
104     if (this.isEmpty()) {
105       return null;
106     }
107     return this.getPointN(this.getNumPoints() - 1);
108   };
109 
110 
111   /**
112    * @return {Boolean} true if LineString is Closed.
113    */
114   jsts.geom.LineString.prototype.isClosed = function() {
115     if (this.isEmpty()) {
116       return false;
117     }
118     return this.getCoordinateN(0).equals2D(
119         this.getCoordinateN(this.components.length - 1));
120   };
121 
122 
123   /**
124    * @return {Boolean} true if LineString is a Ring.
125    */
126   jsts.geom.LineString.prototype.isRing = function() {
127     return this.isClosed() && this.isSimple();
128   };
129 
130 
131   /**
132    * @return {String} String representation of LineString type.
133    */
134   jsts.geom.LineString.prototype.getGeometryType = function() {
135     return 'LineString';
136   };
137 
138 
139   /**
140    * Gets the boundary of this geometry. The boundary of a lineal geometry is
141    * always a zero-dimensional geometry (which may be empty).
142    *
143    * @return {Geometry} the boundary geometry.
144    * @see Geometry#getBoundary
145    */
146   jsts.geom.LineString.prototype.getBoundary = function() {
147     return (new jsts.operation.BoundaryOp(this)).getBoundary();
148   };
149 
150 
151   jsts.geom.LineString.prototype.computeEnvelopeInternal = function() {
152     if (this.isEmpty()) {
153       return new jsts.geom.Envelope();
154     }
155 
156     var env = new jsts.geom.Envelope();
157     this.components.forEach(function(component) {
158       env.expandToInclude(component);
159     });
160 
161     return env;
162   };

165   /**
166    * @param {Geometry}
167    *          other Geometry to compare this LineString to.
168    * @param {double}
169    *          tolerance Tolerance.
170    * @return {Boolean} true if equal.
171    */
172   jsts.geom.LineString.prototype.equalsExact = function(other, tolerance) {
173     if (!this.isEquivalentClass(other)) {
174       return false;
175     }
176 
177     if (this.components.length !== other.components.length) {
178       return false;
179     }
180 
181     if (this.isEmpty() && other.isEmpty()) {
182       return true;
183     }
184 
185     return this.components.reduce(function(equal, component, i) {
186       return equal &&
187           jsts.geom.Geometry.prototype.equal(component, other.components[i],
188               tolerance);
189     });
190   };

193   /**
194    * Creates and returns a full copy of this {@link LineString} object.
195    * (including all coordinates contained by it).
196    *
197    * @return {LineString} a clone of this instance.
198    */
199   jsts.geom.LineString.prototype.clone = function() {
200     var points = this.components.map(function(component) {
201       return component.clone();
202     });
203 
204     var clone = new jsts.geom.LineString(points);
205 
206     return clone;
207   };
208 
209   jsts.geom.LineString.prototype.isEquivalentClass = function(other) {
210     return other instanceof jsts.geom.LineString;
211   };
212 
213   jsts.geom.LineString.prototype.compareToSameClass = function(o) {
214     var line = o;
215     // MD - optimized implementation
216     var i = 0;
217     var j = 0;
218     while (i < this.components.length && j < line.components.length) {
219       var comparison = this.components[i].compareTo(line.components[j]);
220       if (comparison !== 0) {
221         return comparison;
222       }
223       i++;
224       j++;
225     }
226     if (i < this.components.length) {
227       return 1;
228     }
229     if (j < line.components.length) {
230       return -1;
231     }
232     return 0;
233   };
234 
235   jsts.geom.LineString.prototype.apply = function(filter) {
236     if (filter instanceof jsts.geom.GeometryFilter || filter instanceof jsts.geom.GeometryComponentFilter) {
237       filter.filter(this);
238     } else if (filter instanceof jsts.geom.CoordinateFilter) {
239       for (var i = 0; i < this.components.length; i++) {
240         filter.filter(this.components[i]);
241       }
242     }
243   };
244 
245   OpenLayers.Geometry.LineString = jsts.geom.LineString;
