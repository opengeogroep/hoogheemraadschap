 57 /**
 58  * Returns the index of the direction of the point <code>q</code>
 59  * relative to a
 60  * vector specified by <code>p1-p2</code>.
 61  *
 62  * @param {jsts.geom.Coordinate}
 63  *        p1 the origin point of the vector.
 64  * @param {jsts.geom.Coordinate}
 65  *        p2 the final point of the vector.
 66  * @param {jsts.geom.Coordinate}
 67  *        q the point to compute the direction to.
 68  *
 69  * @return {Number}
 70  *         1 if q is counter-clockwise (left) from p1-p2.
 71  * @return {Number}
 72  *         -1 if q is clockwise (right) from p1-p2.
 73  * @return {Number}
 74  *         0 if q is collinear with p1-p2.
 75  */
 76 jsts.algorithm.CGAlgorithms.orientationIndex = function(p1, p2, q) {
 77   /**
 78    * MD - 9 Aug 2010
 79    * It seems that the basic algorithm is slightly orientation dependent,
 80    * when computing the orientation of a point very close to a line.
 81    * This is possibly due to the arithmetic in the translation to the origin.
 82    *
 83    * For instance, the following situation produces identical results
 84    * in spite of the inverse orientation of the line segment:
 85    *
 86    * Coordinate p0 = new Coordinate(219.3649559090992, 140.84159161824724);
 87    * Coordinate p1 = new Coordinate(168.9018919682399, -5.713787599646864);
 88    *
 89    * Coordinate p = new Coordinate(186.80814046338352, 46.28973405831556);
 90    * int orient = orientationIndex(p0, p1, p);
 91    * int orientInv = orientationIndex(p1, p0, p);
 92 
 93    * A way to force consistent results is to normalize the orientation of
 94    * the vector using the following code.
 95    * However, this may make the results of orientationIndex inconsistent
 96    * through the triangle of points, so it's not clear this is
 97    * an appropriate patch.
 98    *
 99    */
100 
101   var dx1, dy1, dx2, dy2;
102   dx1 = p2.x - p1.x;
103   dy1 = p2.y - p1.y;
104   dx2 = q.x - p2.x;
105   dy2 = q.y - p2.y;
106 
107   return jsts.algorithm.RobustDeterminant.signOfDet2x2(dx1, dy1, dx2, dy2);
108 };

157 /**
158  * Tests whether a point lies on the line segments defined by a
159  * list of coordinates.
160  *
161  * @param  {jsts.geom.Coordinate}
162  *         p the coordinate to test.
163  * @param  {Array{jsts.geom.Coordinate}}
164  *         pt An array of coordinates defining line segments
165  * @return {Boolean}
166  *         true if the point is a vertex of the line
167  *         or lies in the interior of a line segment in the linestring.
168  */
169 jsts.algorithm.CGAlgorithms.isOnLine = function(p, pt) {
170   var lineIntersector, i, il, p0, p1;
171   lineIntersector = new jsts.algorithm.RobustLineIntersector();
172 
173   for (i = 1, il = pt.length; i < il; i++) {
174     p0 = pt[i - 1];
175     p1 = pt[i];
176     lineIntersector.computeIntersection(p, p0, p1);
177 
178     if (lineIntersector.hasIntersection()) {
179       return true;
180     }
181   }
182   return false;
183 };

186 /**
187  * Computes whether a ring defined by an array of {@link Coordinate}s is
188  * oriented counter-clockwise.
189  * <ul>
190  * <li>The list of points is assumed to have the first and last points equal.
191  * <li>This will handle coordinate lists which contain repeated points.
192  * </ul>
193  * This algorithm is <b>only</b> guaranteed to work with valid rings.
194  * If the ring is invalid (e.g. self-crosses or touches),
195  * the computed result may not be correct.
196  *
197  * @param {Array{jsts.geom.Coordinate}}
198  *        ring an array of Coordinates forming a ring
199  * @return {Boolean}
200  *         true if the ring is oriented counter-clockwise.
201  * @throws IllegalArgumentException if there are too few points to determine orientation (< 3)
202  */
203 jsts.algorithm.CGAlgorithms.isCCW = function(ring) {
204   var nPts, hiPt, hiIndex, p, iPrev, iNext, prev, next, i, disc, isCCW;
205 
206   //# of points without closing endpoint
207   nPts = ring.length - 1;
208 
209   //sanity check
210   if (nPts < 3) {
211     throw new jsts.IllegalArgumentError('Ring has fewer than 3 points, so orientation cannot be determined');
212   }
213 
214   //find highets point
215   hiPt = ring[0];
216   hiIndex = 0;
217 
218   i = 1;
219   for (i; i <= nPts; i++) {
220     p = ring[i];
221     if (p.y > hiPt.y) {
222       hiPt = p;
223       hiIndex = i;
224     }
225   }
226 
227   //find distinct point before highest point
228   iPrev = hiIndex;
229   do {
230     iPrev = iPrev - 1;
231     if (iPrev < 0) {
232       iPrev = nPts;
233     }
234   }while (ring[iPrev].equals2D(hiPt) && iPrev !== hiIndex);
235 
236   //find distinct point after highest point
237   iNext = hiIndex;
238   do {
239     iNext = (iNext + 1) % nPts;
240   }while (ring[iNext].equals2D(hiPt) && iNext !== hiIndex);
241 
242   prev = ring[iPrev];
243   next = ring[iNext];
244 
245   /**
246    * This check catches cases where the ring contains an A-B-A configuration of points.
247    * This can happen if the ring does not contain 3 distinct points
248    * (including the case where the input array has fewer than 4 elements),
249    * or it contains coincident line segments.
250    */
251   if (prev.equals2D(hiPt) || next.equals2D(hiPt) || prev.equals2D(next)) {
252     return false;
253   }
254 
255   disc = jsts.algorithm.CGAlgorithms.computeOrientation(prev, hiPt, next);
256 
257   /**
258    *  If disc is exactly 0, lines are collinear.  There are two possible cases:
259    *  (1) the lines lie along the x axis in opposite directions
260    *  (2) the lines lie on top of one another
261    *
262    *  (1) is handled by checking if next is left of prev ==> CCW
263    *  (2) will never happen if the ring is valid, so don't check for it
264    *  (Might want to assert this)
265    */
266   isCCW = false;
267   if (disc === 0) {
268     // poly is CCW if prev x is right of next x
269     isCCW = (prev.x > next.x);
270   }
271   else {
272     // if area is positive, points are ordered CCW
273     isCCW = (disc > 0);
274   }
275 
276   return isCCW;
277 };

304 /**
305  * Computes the distance from a point p to a line segment AB
306  *
307  * Note: NON-ROBUST!
308  *
309  * @param {jsts.geom.Coordinate}
310  *        p the point to compute the distance for.
311  * @param {jsts.geom.Coordinate}
312  *        A one point of the line.
313  * @param {jsts.geom.Coordinate}
314  *        B another point of the line (must be different to A).
315  * @return {Number}
316  *         the distance from p to line segment AB.
317  */
318 jsts.algorithm.CGAlgorithms.distancePointLine = function(p, A, B) {
319   if (!(A instanceof jsts.geom.Coordinate)) {
320     jsts.algorithm.CGAlgorithms.distancePointLine2.apply(this, arguments);
321   }
322 
323   //if start = end, then just compute distance to one of the endpoints
324   if (A.x === B.x && A.y === B.y) {
325     return p.distance(A);
326   }
327   //otherwise use comp.graphics.algorithms Frequently Asked Questions method
328   /*(1)             AC dot AB
329                    r = ---------
330                          ||AB||^2
331     r has the following meaning:
332     r=0 P = A
333     r=1 P = B
334     r<0 P is on the backward extension of AB
335     r>1 P is on the forward extension of AB
336     0<r<1 P is interior to AB
337   */
338   var r, s;
339   r = ((p.x - A.x) * (B.x - A.x) + (p.y - A.y) * (B.y - A.y)) /
340       ((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y));
341 
342   if (r <= 0.0) {
343     return p.distance(A);
344   }
345   if (r >= 1.0) {
346     return p.distance(B);
347   }
348 
349   /*(2)
350     (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
351   s = -----------------------------
352              L^2
353 
354   Then the distance from C to P = |s|*L.
355   */
356 
357   s = ((A.y - p.y) * (B.x - A.x) - (A.x - p.x) * (B.y - A.y)) /
358       ((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y));
359 
360   return Math.abs(s) *
361       Math.sqrt(((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y)));
362 };

365 /**
366  * Computes the perpendicular distance from a point p
367  * to the (infinite) line containing the points AB
368  *
369  * @param {jsts.geom.Coordinate}
370  *        p the point to compute the distance for.
371  * @param {jsts.geom.Coordinate}
372  *        A one point of the line.
373  * @param {jsts.geom.Coordinate}
374  *        B another point of the line (must be different to A).
375  * @return {Number}
376  *         the distance from p to line AB.
377  */
378 jsts.algorithm.CGAlgorithms.distancePointLinePerpendicular = function(p, A, B) {
379   //use comp.graphics.algorithms Frequently Asked Questions method
380   /*(2)
381                    (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
382               s = -----------------------------
383                                    L^2
384 
385               Then the distance from C to P = |s|*L.
386   */
387   var s = ((A.y - p.y) * (B.x - A.x) - (A.x - p.x) * (B.y - A.y)) /
388       ((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y));
389 
390   return Math.abs(s) *
391       Math.sqrt(((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y)));
392 };

395 /**
396  * Computes the distance from a point to a sequence
397  * of line segments.
398  *
399  * @param {jsts.geom.Coordinate}
400  *        p a point.
401  * @param {Array{jsts.geom.Coordinate}}
402  *        line a sequence of contiguous line segments defined by their vertices
403  * @return {Number}
404  *         the minimum distance between the point and the line segments.
405  */
406 jsts.algorithm.CGAlgorithms.distancePointLine2 = function(p, line) {
407   var minDistance, i, il, dist;
408   if (line.length === 0) {
409     throw new jsts.error.IllegalArgumentError('Line array must contain at least one vertex');
410   }
411   minDistance = p.distance(line[0]);
412   for (i = 0, il = line.length - 1; i < il; i++) {
413     dist = jsts.algorithm.CGAlgorithms.distancePointLine(p, line[i], line[i + 1]);
414     if (dist < minDistance) {
415       minDistance = dist;
416     }
417   }
418   return minDistance;
419 };

421 /**
422  * Computes the distance from a line segment AB to a line segment CD
423  *
424  * Note: NON-ROBUST!
425  *
426  * @param {jsts.geom.Coordinate}
427  *        A a point of one line.
428  * @param {jsts.geom.Coordinate}
429  *        B the second point of  (must be different to A).
430  * @param {jsts.geom.Coordinate}
431  *        C one point of the line.
432  * @param {jsts.geom.Coordinate}
433  *        D another point of the line (must be different to A).
434  * @return {Number}
435  *         the distance.
436  */
437 
438 jsts.algorithm.CGAlgorithms.distanceLineLine = function(A, B, C, D) {
439   //check for zero-length segments
440   if (A.equals(B)) {
441     return jsts.algorithm.CGAlgorithms.distancePointLine(A, C, D);
442   }
443   if (C.equals(D)) {
444     return jsts.algorithm.CGAlgorithms.distancePointLine(D, A, B);
445   }
446 
447   //AB and CD are line segments
448   /* from comp.graphics.algo
449 
450   Solving the above for r and s yields
451         (Ay-Cy)(Dx-Cx)-(Ax-Cx)(Dy-Cy)
452              r = ----------------------------- (eqn 1)
453         (Bx-Ax)(Dy-Cy)-(By-Ay)(Dx-Cx)
454 
455       (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
456     s = ----------------------------- (eqn 2)
457       (Bx-Ax)(Dy-Cy)-(By-Ay)(Dx-Cx)
458   Let P be the position vector of the intersection point, then
459     P=A+r(B-A) or
460     Px=Ax+r(Bx-Ax)
461     Py=Ay+r(By-Ay)
462   By examining the values of r & s, you can also determine some other
463   limiting conditions:
464     If 0<=r<=1 & 0<=s<=1, intersection exists
465     r<0 or r>1 or s<0 or s>1 line segments do not intersect
466     If the denominator in eqn 1 is zero, AB & CD are parallel
467     If the numerator in eqn 1 is also zero, AB & CD are collinear.
468 
469   */
470   var r_top, r_bot, s_top, s_bot, s, r;
471   r_top = (A.y - C.y) * (D.x - C.x) - (A.x - C.x) * (D.y - C.y);
472   r_bot = (B.x - A.x) * (D.y - C.y) - (B.y - A.y) * (D.x - C.x);
473 
474   s_top = (A.y - C.y) * (B.x - A.x) - (A.x - C.x) * (B.y - A.y);
475   s_bot = (B.x - A.x) * (D.y - C.y) - (B.y - A.y) * (D.x - C.x);
476 
477 
478   if ((r_bot === 0) || (s_bot === 0)) {
479     return Math.min(jsts.algorithm.CGAlgorithms.distancePointLine(A, C, D),
480         Math.min(jsts.algorithm.CGAlgorithms.distancePointLine(B, C, D),
481             Math.min(jsts.algorithm.CGAlgorithms.distancePointLine(C, A, B),
482                 jsts.algorithm.CGAlgorithms.distancePointLine(D, A, B))));
483   }
484 
485   s = s_top / s_bot;
486   r = r_top / r_bot;
487   if ((r < 0) || (r > 1) || (s < 0) || (s > 1)) {
488     //no intersection
489     return Math.min(jsts.algorithm.CGAlgorithms.distancePointLine(A, C, D),
490         Math.min(jsts.algorithm.CGAlgorithms.distancePointLine(B, C, D),
491         Math.min(jsts.algorithm.CGAlgorithms.distancePointLine(C, A, B),
492             jsts.algorithm.CGAlgorithms.distancePointLine(D, A, B))));
493   }
494 
495   return 0.0; //intersection exists
496 };

499 /**
500  * Computes the signed area for a ring.
501  * The signed area is positive if
502  * the ring is oriented CW, negative if the ring is oriented CCW,
503  * and zero if the ring is degenerate or flat.
504  *
505  * @param {Array{jsts.geom.Coordinate}}
506  *        ring the coordinates forming the ring
507  * @return {Number}
508  *         the signed area of the ring.
509  */
510 jsts.algorithm.CGAlgorithms.signedArea = function(ring) {
511   if (ring.length < 3) {
512     return 0.0;
513   }
514   var sum, i, il, bx, by, cx, cy;
515 
516   sum = 0.0;
517 
518   for (i = 0, il = ring.length - 1; i < il; i++) {
519     bx = ring[i].x;
520     by = ring[i].y;
521     cx = ring[i + 1].x;
522     cy = ring[i + 1].y;
523     sum += (bx + cx) * (cy - by);
524   }
525 
526   return -sum / 2.0;
527 };

530 /**
531  * Computes the signed area for a ring.
532  * The signed area is:
533  * <ul>
534  * <li>positive if the ring is oriented CW
535  * <li>negative if the ring is oriented CCW
536  * <li>zero if the ring is degenerate or flat
537  * </ul>
538  *
539  * @param {Array{jsts.geom.Coordinate}}
540  *        ring the coordinates forming the ring
541  * @return {Number}
542  *         the signed area of the ring.
543  */
544 jsts.algorithm.CGAlgorithms.signedArea = function(ring) {
545   var n, sum, p, bx, by, i, cx, cy;
546 
547   n = ring.length;
548   if (n < 3) {
549     return 0.0;
550   }
551 
552   sum = 0.0;
553   p = ring[0];
554 
555   bx = p.x;
556   by = p.y;
557 
558   for (i = 1; i < n; i++) {
559     p = ring[i];
560     cx = p.x;
561     cy = p.y;
562     sum += (bx + cx) * (cy - by);
563     bx = cx;
564     by = cy;
565   }
566 
567   return -sum / 2.0;
568 };
