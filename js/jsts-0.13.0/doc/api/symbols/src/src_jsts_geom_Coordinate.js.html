  7 (function() {
  8 
  9   /**
 10    * A lightweight class used to store coordinates on the 2-dimensional
 11    * Cartesian plane. It is distinct from {@link Point}, which is a subclass of
 12    * {@link Geometry}. Unlike objects of type {@link Point} (which contain
 13    * additional information such as an envelope, a precision model, and spatial
 14    * reference system information), a <code>Coordinate</code> only contains
 15    * coordinate values and accessor methods.
 16    *
 17    * @requires jsts/geom/Geometry.js
 18    */
 19 
 20 
 21 
 22   /**
 23    * @constructor
 24    */
 25   jsts.geom.Coordinate = function(x, y) {
 26     OpenLayers.Geometry.prototype.initialize.apply(this, arguments);
 27     if (x === undefined || x === null) {
 28       this.x = 0;
 29       this.y = 0;
 30     } else if (typeof x === 'number' || typeof x === 'string') {
 31       this.x = parseFloat(x);
 32       this.y = parseFloat(y);
 33     } else if (x instanceof jsts.geom.Coordinate) {
 34       y = x.y;
 35       x = x.x;
 36       this.x = parseFloat(x);
 37       this.y = parseFloat(y);
 38     }
 39   };
 40   jsts.geom.Coordinate.prototype = OpenLayers.Geometry.Point.prototype;
 41 
 42 
 43   /**
 44    * Sets this <code>Coordinate</code>s (x,y,z) values to that of
 45    * <code>other</code>.
 46    *
 47    * @param {Coordinate}
 48    *          other the <code>Coordinate</code> to copy.
 49    */
 50   jsts.geom.Coordinate.prototype.setCoordinate = function(other) {
 51     this.x = other.x;
 52     this.y = other.y;
 53   };
 54 
 55 
 56   /**
 57    * Clones this instance.
 58    *
 59    * @return {Coordinate} A point instance cloned from this.
 60    */
 61   jsts.geom.Coordinate.prototype.clone = function() {
 62     return new jsts.geom.Coordinate(this.x, this.y);
 63   };
 64 
 65 
 66   /**
 67    * Computes the 2-dimensional Euclidean distance to another location. The
 68    * Z-ordinate is ignored.
 69    *
 70    * @param {Coordinate}
 71    *          p a point.
 72    * @return {number} the 2-dimensional Euclidean distance between the
 73    *         locations.
 74    */
 75   jsts.geom.Coordinate.prototype.distance = function(p) {
 76     var dx = this.x - p.x;
 77     var dy = this.y - p.y;
 78 
 79     return Math.sqrt(dx * dx + dy * dy);
 80   };
 81 
 82   /**
 83    * Returns whether the planar projections of the two <code>Coordinate</code>s
 84    * are equal.
 85    *
 86    * @param {Coordinate}
 87    *          other a <code>Coordinate</code> with which to do the 2D
 88    *          comparison.
 89    * @return {boolean} <code>true</code> if the x- and y-coordinates are
 90    *         equal; the z-coordinates do not have to be equal.
 91    */
 92   jsts.geom.Coordinate.prototype.equals2D = function(other) {
 93     if (this.x !== other.x) {
 94       return false;
 95     }
 96 
 97     if (this.y !== other.y) {
 98       return false;
 99     }
100 
101     return true;
102   };
103 
104   /**
105    * Returns <code>true</code> if <code>other</code> has the same values for
106    * the x and y ordinates. Since Coordinates are 2.5D, this routine ignores the
107    * z value when making the comparison.
108    *
109    * @param {Coordinate}
110    *          other a <code>Coordinate</code> with which to do the comparison.
111    * @return {boolean} <code>true</code> if <code>other</code> is a
112    *         <code>Coordinate</code> with the same values for the x and y
113    *         ordinates.
114    */
115   jsts.geom.Coordinate.prototype.equals = function(other) {
116     if (!other instanceof jsts.geom.Coordinate || other === undefined) {
117       return false;
118     }
119     return this.equals2D(other);
120   };
121 
122   /**
123    * Compares this {@link Coordinate} with the specified {@link Coordinate} for
124    * order. This method ignores the z value when making the comparison. Returns:
125    * <UL>
126    * <LI> -1 : this.x < other.x || ((this.x == other.x) && (this.y < other.y))
127    * <LI> 0 : this.x == other.x && this.y = other.y
128    * <LI> 1 : this.x > other.x || ((this.x == other.x) && (this.y > other.y))
129    *
130    * </UL>
131    * Note: This method assumes that ordinate values are valid numbers. NaN
132    * values are not handled correctly.
133    *
134    * @param {Coordinate}
135    *          other the <code>Coordinate</code> with which this
136    *          <code>Coordinate</code> is being compared.
137    * @return {number} -1, zero, or 1 as explained above.
138    */
139   jsts.geom.Coordinate.prototype.compareTo = function(other) {
140     if (this.x < other.x) {
141       return -1;
142     }
143     if (this.x > other.x) {
144       return 1;
145     }
146     if (this.y < other.y) {
147       return -1;
148     }
149     if (this.y > other.y) {
150       return 1;
151     }
152 
153     return 0;
154   };
155 
156   jsts.geom.Coordinate.prototype.hashCode = function() {
157     return '' + this.x + this.y;
158   };
159 
160   /**
161    * @private
162    */
163   OpenLayers.Geometry.Point = function(x, y) {
164     OpenLayers.Geometry.prototype.initialize.apply(this, arguments);
165     if (x === undefined || x === null) {
166       this.x = 0;
167       this.y = 0;
168     } else if (typeof x === 'number' || typeof x === 'string') {
169       this.x = parseFloat(x);
170       this.y = parseFloat(y);
171     } else if (x instanceof jsts.geom.Coordinate) {
172       y = x.y;
173       x = x.x;
174       this.x = parseFloat(x);
175       this.y = parseFloat(y);
176     }
177   };
178 
179   /**
180    * @private
181    */
182   OpenLayers.Geometry.Point = jsts.geom.Coordinate;
183 
184 })();
