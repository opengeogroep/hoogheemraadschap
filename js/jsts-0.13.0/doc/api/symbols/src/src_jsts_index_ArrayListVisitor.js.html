  9 /**
 10  * An array-visitor
 11  *
 12  * Initializes this class with the openlayers inheritance mechanism
 13  *
 14  * @constructor
 15  */
 16 jsts.index.ArrayListVisitor = function() {
 17   this.items = [];
 18 };
 19 
 20 
 21 /**
 22  * Visits an item
 23  *
 24  * @param {Object}
 25  *          item the item to visit.
 26  */
 27 jsts.index.ArrayListVisitor.prototype.visitItem = function(item) {
 28   this.items.push(item);
 29 };
 30 
 31 
 32 /**
 33  * Returns all visited items
 34  *
 35  * @return {Array} An array with all visited items.
 36  */
 37 jsts.index.ArrayListVisitor.prototype.getItems = function() {
 38   return this.items;
 39 };
